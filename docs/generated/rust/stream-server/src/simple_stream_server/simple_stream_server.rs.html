50async fn main() -> Result<(), Box<dyn std::error::Error>> {
51    // Configuration du logging
52    tracing_subscriber::fmt::init();
53    info!("üéµ D√©marrage du Stream Server Simplifi√©");
54
55    // Configuration
56    let audio_dir = std::env::var("AUDIO_DIR")
57        .unwrap_or_else(|_| "/opt/veza-stream/audio".to_string());
58    let port = std::env::var("PORT")
59        .unwrap_or_else(|_| "8000".to_string())
60        .parse::<u16>()
61        .unwrap_or(8000);
62
63    // Cr√©er le r√©pertoire audio s'il n'existe pas
64    tokio::fs::create_dir_all(&audio_dir).await?;
65    
66    let state = AppState {
67        audio_dir: PathBuf::from(audio_dir.clone()),
68        port,
69    };
70
71    info!("üìÅ R√©pertoire audio: {}", audio_dir);
72    info!("üîå Port: {}", port);
73
74    // Configuration CORS permissive
75    let cors = CorsLayer::new()
76        .allow_origin(Any)
77        .allow_methods([axum::http::Method::GET, axum::http::Method::OPTIONS])
78        .allow_headers(Any);
79
80    // Routes
81    let app = Router::new()
82        .route("/", get(root))
83        .route("/health", get(health_check))
84        .route("/stream/:filename", get(stream_audio))
85        .route("/info/:filename", get(audio_info))
86        .route("/list", get(list_audio_files))
87        .layer(CompressionLayer::new())
88        .layer(cors)
89        .with_state(state);
90
91    // D√©marrage du serveur
92    let addr = SocketAddr::from(([0, 0, 0, 0], port));
93    info!("üåê Serveur d√©marr√© sur http://{}", addr);
94
95    let listener = tokio::net::TcpListener::bind(&addr).await?;
96    axum::serve(listener, app)
97        .with_graceful_shutdown(shutdown_signal())
98        .await?;
99
100    info!("üëã Serveur arr√™t√©");
101    Ok(())
102}

120async fn list_audio_files(State(state): State<AppState>) -> Result<Json<Vec<String>>, StatusCode> {
121    let mut files = Vec::new();
122    
123    let mut dir = match fs::read_dir(&state.audio_dir).await {
124        Ok(dir) => dir,
125        Err(_) => return Ok(Json(files)),
126    };
127
128    while let Some(entry) = dir.next_entry().await.unwrap_or(None) {
129        if let Some(name) = entry.file_name().to_str() {
130            if name.ends_with(".mp3") || name.ends_with(".wav") || name.ends_with(".flac") {
131                files.push(name.to_string());
132            }
133        }
134    }
135
136    files.sort();
137    Ok(Json(files))
138}

140async fn audio_info(
141    Path(filename): Path<String>,
142    State(state): State<AppState>,
143) -> Result<Json<StreamInfo>, StatusCode> {
144    let file_path = state.audio_dir.join(&filename);
145    
146    if !file_path.exists() {
147        return Err(StatusCode::NOT_FOUND);
148    }
149
150    let metadata = match fs::metadata(&file_path).await {
151        Ok(meta) => meta,
152        Err(_) => return Err(StatusCode::INTERNAL_SERVER_ERROR),
153    };
154
155    let content_type = match file_path.extension().and_then(|s| s.to_str()) {
156        Some("mp3") => "audio/mpeg",
157        Some("wav") => "audio/wav",
158        Some("flac") => "audio/flac",
159        _ => "application/octet-stream",
160    };
161
162    Ok(Json(StreamInfo {
163        filename,
164        size: Some(metadata.len()),
165        content_type: content_type.to_string(),
166    }))
167}

169async fn stream_audio(
170    Path(filename): Path<String>,
171    Query(_params): Query<StreamParams>,
172    State(state): State<AppState>,
173    headers: HeaderMap,
174) -> Result<axum::response::Response, StatusCode> {
175    let file_path = state.audio_dir.join(&filename);
176    
177    if !file_path.exists() {
178        warn!("Fichier non trouv√©: {:?}", file_path);
179        return Err(StatusCode::NOT_FOUND);
180    }
181
182    let file = match fs::File::open(&file_path).await {
183        Ok(file) => file,
184        Err(e) => {
185            error!("Erreur ouverture fichier: {}", e);
186            return Err(StatusCode::INTERNAL_SERVER_ERROR);
187        }
188    };
189
190    let metadata = match file.metadata().await {
191        Ok(meta) => meta,
192        Err(_) => return Err(StatusCode::INTERNAL_SERVER_ERROR),
193    };
194
195    let file_size = metadata.len();
196    let content_type = match file_path.extension().and_then(|s| s.to_str()) {
197        Some("mp3") => "audio/mpeg",
198        Some("wav") => "audio/wav",
199        Some("flac") => "audio/flac",
200        _ => "application/octet-stream",
201    };
202
203    // Support pour les requ√™tes Range (streaming partiel)
204    if let Some(range) = headers.get(header::RANGE) {
205        if let Ok(range_str) = range.to_str() {
206            if let Some(range_value) = parse_range_header(range_str, file_size) {
207                let (start, end) = range_value;
208                let content_length = end - start + 1;
209                
210                info!("Streaming partiel: {}-{}/{} pour {}", start, end, file_size, filename);
211                
212                let stream = create_partial_stream(file, start, content_length).await?;
213                
214                return Ok(axum::response::Response::builder()
215                    .status(StatusCode::PARTIAL_CONTENT)
216                    .header(header::CONTENT_TYPE, content_type)
217                    .header(header::CONTENT_LENGTH, content_length.to_string())
218                    .header(header::CONTENT_RANGE, format!("bytes {}-{}/{}", start, end, file_size))
219                    .header(header::ACCEPT_RANGES, "bytes")
220                    .body(axum::body::Body::from_stream(stream))
221                    .unwrap());
222            }
223        }
224    }
225
226    // Streaming complet
227    info!("Streaming complet: {} ({} bytes)", filename, file_size);
228    let stream = create_full_stream(file).await?;
229    
230    Ok(axum::response::Response::builder()
231        .status(StatusCode::OK)
232        .header(header::CONTENT_TYPE, content_type)
233        .header(header::CONTENT_LENGTH, file_size.to_string())
234        .header(header::ACCEPT_RANGES, "bytes")
235        .body(axum::body::Body::from_stream(stream))
236        .unwrap())
237}

239fn parse_range_header(range: &str, file_size: u64) -> Option<(u64, u64)> {
240    if !range.starts_with("bytes=") {
241        return None;
242    }
243    
244    let range = &range[6..];
245    let parts: Vec<&str> = range.split('-').collect();
246    
247    if parts.len() != 2 {
248        return None;
249    }
250    
251    let start = if parts[0].is_empty() {
252        file_size.saturating_sub(parts[1].parse::<u64>().ok()?)
253    } else {
254        parts[0].parse::<u64>().ok()?
255    };
256    
257    let end = if parts[1].is_empty() {
258        file_size - 1
259    } else {
260        parts[1].parse::<u64>().ok()?.min(file_size - 1)
261    };
262    
263    if start <= end && end < file_size {
264        Some((start, end))
265    } else {
266        None
267    }
268}

277async fn create_partial_stream(
278    mut file: tokio::fs::File,
279    start: u64,
280    length: u64,
281) -> Result<impl futures_util::Stream<Item = Result<bytes::Bytes, std::io::Error>>, StatusCode> {
282    use tokio::io::{AsyncSeekExt, SeekFrom};
283    use tokio_util::io::ReaderStream;
284    
285    if let Err(_) = file.seek(SeekFrom::Start(start)).await {
286        return Err(StatusCode::INTERNAL_SERVER_ERROR);
287    }
288    
289    let limited_reader = file.take(length);
290    Ok(ReaderStream::new(limited_reader))
291}

293async fn shutdown_signal() {
294    let ctrl_c = async {
295        signal::ctrl_c()
296            .await
297            .expect("Impossible d'installer le handler Ctrl+C");
298    };
299
300    #[cfg(unix)]
301    let terminate = async {
302        signal::unix::signal(signal::unix::SignalKind::terminate())
303            .expect("Impossible d'installer le handler SIGTERM")
304            .recv()
305            .await;
306    };
307
308    #[cfg(not(unix))]
309    let terminate = std::future::pending::<()>();
310
311    tokio::select! {
312        _ = ctrl_c => {
313            info!("üì± Signal Ctrl+C re√ßu, arr√™t du serveur...");
314        },
315        _ = terminate => {
316            info!("üì± Signal SIGTERM re√ßu, arr√™t du serveur...");
317        }
318    }
319}

165            Self::Ms10 => 10.0,

212            channels: config.channels,

214            complexity: config.complexity,

217                crate::codecs::LatencyMode::UltraLow => OpusApplication::RestrictedLowDelay,
218                _ => OpusApplication::Audio,

221            frame_duration: match config.latency_mode {
222                crate::codecs::LatencyMode::UltraLow => OpusFrameDuration::Ms2_5,
223                crate::codecs::LatencyMode::Low => OpusFrameDuration::Ms5,
224                _ => OpusFrameDuration::Ms10,
225            },
226            packet_loss_resilience: true,
227            dtx_enabled: false,
228            inband_fec: true,
229        };
230        
231        let frame_size = opus_config.frame_duration.frame_size(opus_config.sample_rate) * opus_config.channels as usize;
232        
233        let encoder_state = OpusEncoderState {
234            initialized: false,
235            last_frame_time: None,
236            total_frames: 0,
237            total_bytes: 0,
238            last_bitrate: opus_config.bitrate,
239            bandwidth_adaptation: BandwidthAdaptation {
240                current_bitrate: opus_config.bitrate,
241                target_bitrate: opus_config.bitrate,
242                _adaptation_rate: 0.1,
243                _quality_history: Vec::new(),
244            },
245        };
246        
247        let mut encoder = Self {
248            config: opus_config,
249            encoder_state: Arc::new(Mutex::new(encoder_state)),
250            metrics: EncoderMetrics::default(),

253        };

279    /// Valide la configuration
280    fn validate_config(&self) -> Result<(), AppError> {
281        // Opus supporte spécifiquement 8kHz, 12kHz, 16kHz, 24kHz, 48kHz
282        let valid_rates = [8000, 12000, 16000, 24000, 48000];
283        if !valid_rates.contains(&self.config.sample_rate) {
284            return Err(AppError::InvalidSampleRate {
285                rate: self.config.sample_rate,
286            });
287        }
288        
289        // Opus supporte 1 ou 2 channels
290        if self.config.channels == 0 || self.config.channels > 2 {
291            return Err(AppError::InvalidChannelCount {
292                channels: self.config.channels,
293            });
294        }
295        
296        // Bitrate valide : 6kbps à 512kbps
297        if self.config.bitrate < 6_000 || self.config.bitrate > 512_000 {
298            return Err(AppError::InvalidBitrate {
299                bitrate: self.config.bitrate,
300                codec: "opus".to_string(),
301            });
302        }
303        
304        Ok(())
305    }

307    /// Adapte automatiquement le bitrate selon les conditions
308    #[allow(dead_code)]
309    fn adapt_bitrate(&mut self, available_bandwidth: u32, packet_loss: f32) -> Result<(), AppError> {
310        let mut state = self.encoder_state.lock();
311        
312        // Calcul du bitrate optimal
313        let mut target_bitrate = available_bandwidth * 8 / 10; // 80% de la bande passante
314        
315        // Réduction si perte de paquets
316        if packet_loss > 0.01 { // > 1%
317            target_bitrate = (target_bitrate as f32 * (1.0 - packet_loss * 2.0)) as u32;
318        }
319        
320        // Limites du codec
321        target_bitrate = target_bitrate.clamp(6_000, 512_000);
322        
323        if target_bitrate != state.bandwidth_adaptation.current_bitrate {
324            state.bandwidth_adaptation.target_bitrate = target_bitrate;
325            
326            // Adaptation progressive
327            let diff = target_bitrate as f32 - state.bandwidth_adaptation.current_bitrate as f32;
328            let adjustment = diff * state.bandwidth_adaptation._adaptation_rate;
329            state.bandwidth_adaptation.current_bitrate = 
330                (state.bandwidth_adaptation.current_bitrate as f32 + adjustment) as u32;
331            
332            debug!("Adaptation bitrate Opus: {} -> {} bps", 
333                   state.last_bitrate, state.bandwidth_adaptation.current_bitrate);
334        }
335        
336        Ok(())
337    }

339    /// Encode avec détection automatique du type de signal
340    fn encode_with_signal_detection(&mut self, samples: &[f32]) -> Result<Vec<u8>, AppError> {
341        // Analyse simple du signal pour optimiser l'encodage
342        let signal_type = self.detect_signal_type(samples);
343        
344        // Simulation d'encodage Opus
345        // En production, utiliser opus_encode() ou opus_encode_float()
346        let encoded_size = self.estimate_encoded_size(samples.len());
347        let mut encoded_data = vec![0u8; encoded_size];
348        
349        // Simulation de l'encodage avec compression réaliste
350        self.simulate_opus_encoding(samples, &mut encoded_data)?;
351        
352        // Mise à jour des métriques
353        self.update_encoding_metrics(samples.len(), encoded_data.len());
354        
355        debug!("Frame Opus encodée: {} samples -> {} bytes (type: {:?})", 
356               samples.len(), encoded_data.len(), signal_type);
357        
358        Ok(encoded_data)
359    }

361    /// Détecte le type de signal pour optimiser l'encodage
362    fn detect_signal_type(&self, samples: &[f32]) -> OpusSignalType {
363        if samples.is_empty() {
364            return OpusSignalType::Auto;
365        }
366        
367        // Calcul de métriques simples
368        let energy = samples.iter().map(|&s| s * s).sum::<f32>() / samples.len() as f32;
369        let zero_crossings = samples.windows(2)
370            .filter(|w| (w[0] >= 0.0) != (w[1] >= 0.0))
371            .count();
372        
373        let zcr = zero_crossings as f32 / samples.len() as f32;
374        
375        // Heuristiques simples
376        if zcr > 0.1 && energy > 0.01 {
377            OpusSignalType::Voice
378        } else if energy > 0.001 {
379            OpusSignalType::Music
380        } else {
381            OpusSignalType::Auto
382        }
383    }

385    /// Estime la taille encodée
386    fn estimate_encoded_size(&self, sample_count: usize) -> usize {
387        let duration_ms = (sample_count as f32 / self.config.sample_rate as f32) * 1000.0;
388        let bits_per_ms = self.config.bitrate as f32 / 1000.0;
389        let estimated_bits = duration_ms * bits_per_ms;
390        (estimated_bits / 8.0) as usize
391    }

393    /// Simulation d'encodage Opus
394    fn simulate_opus_encoding(&self, samples: &[f32], output: &mut [u8]) -> Result<(), AppError> {
395        // Simulation simple - en production, utiliser libopus
396        for (i, &sample) in samples.iter().enumerate() {
397            if i / 4 < output.len() {
398                // Quantification simulée
399                let quantized = (sample * 127.0) as i8;
400                output[i / 4] = quantized as u8;
401            }
402        }
403        Ok(())
404    }

406    /// Met à jour les métriques d'encodage
407    fn update_encoding_metrics(&mut self, input_samples: usize, output_bytes: usize) {
408        self.metrics.frames_encoded += 1;
409        self.metrics.bytes_output += output_bytes as u64;
410        
411        // Calcul du ratio de compression
412        let input_bytes = input_samples * 4; // f32 = 4 bytes
413        self.metrics.compression_ratio = input_bytes as f32 / output_bytes as f32;
414        
415        // Mise à jour de l'état
416        {
417            let mut state = self.encoder_state.lock();
418            state.total_frames += 1;
419            state.total_bytes += output_bytes as u64;
420            state.last_frame_time = Some(Instant::now());
421        }
422    }

425impl AudioEncoder for OpusEncoderImpl {
426    fn encode(&mut self, samples: &[f32], sample_rate: u32, channels: u8) -> Result<Vec<u8>, AppError> {
427        // Vérifier que les paramètres correspondent à la config
428        if sample_rate != self.config.sample_rate || channels != self.config.channels {
429            return Err(AppError::ParameterMismatch {
430                expected: format!("{}Hz, {} ch", self.config.sample_rate, self.config.channels),
431                got: format!("{}Hz, {} ch", sample_rate, channels),
432            });
433        }
434        
435        // Ajouter les échantillons au buffer
436        self.sample_buffer.extend_from_slice(samples);
437        
438        let mut encoded_frames = Vec::new();
439        
440        // Traiter les frames complètes
441        while self.sample_buffer.len() >= self.frame_size {
442            let frame_samples: Vec<f32> = self.sample_buffer.drain(..self.frame_size).collect();
443            let encoded_frame = self.encode_with_signal_detection(&frame_samples)?;
444            encoded_frames.extend(encoded_frame);
445        }
446        
447        Ok(encoded_frames)
448    }
449    
450    fn finalize(&mut self) -> Result<Vec<u8>, AppError> {
451        // Encoder le buffer restant avec padding si nécessaire
452        if !self.sample_buffer.is_empty() {
453            // Padding avec du silence
454            while self.sample_buffer.len() < self.frame_size {
455                self.sample_buffer.push(0.0);
456            }
457            
458            let final_frame: Vec<f32> = self.sample_buffer.drain(..).collect();
459            return self.encode_with_signal_detection(&final_frame);
460        }
461        
462        Ok(Vec::new())
463    }
464    
465    fn reset(&mut self) -> Result<(), AppError> {
466        self.sample_buffer.clear();
467        self.metrics = EncoderMetrics::default();
468        
469        {
470            let mut state = self.encoder_state.lock();
471            state.total_frames = 0;
472            state.total_bytes = 0;
473            state.last_frame_time = None;
474        }
475        
476        debug!("Encodeur Opus remis à zéro");
477        Ok(())
478    }
479    
480    fn set_bitrate(&mut self, bitrate: u32) -> Result<(), AppError> {
481        if bitrate < 6_000 || bitrate > 512_000 {
482            return Err(AppError::InvalidBitrate {
483                bitrate,
484                codec: "opus".to_string(),
485            });
486        }
487        
488        self.config.bitrate = bitrate;
489        
490        {
491            let mut state = self.encoder_state.lock();
492            state.last_bitrate = bitrate;
493            state.bandwidth_adaptation.current_bitrate = bitrate;
494            state.bandwidth_adaptation.target_bitrate = bitrate;
495        }
496        
497        debug!("Bitrate Opus mis à jour: {} bps", bitrate);
498        Ok(())
499    }

565    /// Décode avec dissimulation de perte de paquets
566    fn decode_with_plc(&mut self, data: Option<&[u8]>) -> Result<DecodedAudio, AppError> {
567        let mut state = self.decoder_state.lock();
568        
569        match data {
570            Some(encoded_data) => {
571                // Décodage normal
572                let decoded_samples = self.simulate_opus_decoding(encoded_data)?;
573                
574                // Sauvegarder pour PLC
575                state.packet_loss_concealment.last_good_frame = Some(decoded_samples.clone());
576                state.packet_loss_concealment.consecutive_losses = 0;
577                state.total_frames += 1;
578                state.last_decode_time = Some(Instant::now());
579                
580                Ok(DecodedAudio {
581                    samples: decoded_samples,
582                    sample_rate: self.config.sample_rate,
583                    channels: self.config.channels,
584                    duration_ms: self.config.frame_duration.as_ms() as u32,
585                    format: AudioSampleFormat::F32,
586                })
587            }
588            None => {
589                // Perte de paquet - générer du contenu de remplacement
590                state.packet_loss_concealment.consecutive_losses += 1;
591                
592                let concealed_samples = self.generate_packet_loss_concealment(&mut state)?;
593                
594                Ok(DecodedAudio {
595                    samples: concealed_samples,
596                    sample_rate: self.config.sample_rate,
597                    channels: self.config.channels,
598                    duration_ms: self.config.frame_duration.as_ms() as u32,
599                    format: AudioSampleFormat::F32,
600                })
601            }
602        }
603    }
604    
605    /// Simulation de décodage Opus
606    fn simulate_opus_decoding(&self, data: &[u8]) -> Result<Vec<f32>, AppError> {
607        let frame_size = self.config.frame_duration.frame_size(self.config.sample_rate) * self.config.channels as usize;
608        let mut samples = Vec::with_capacity(frame_size);
609        
610        // Simulation simple - en production, utiliser opus_decode()
611        for (i, &byte) in data.iter().enumerate() {
612            if samples.len() < frame_size {
613                // Dé-quantification simulée
614                let sample = (byte as i8) as f32 / 127.0;
615                samples.push(sample);
616                
617                // Interpolation pour atteindre la taille de frame
618                if i < data.len() - 1 {
619                    for _ in 0..3 {
620                        if samples.len() < frame_size {
621                            samples.push(sample * 0.8); // Interpolation simple
622                        }
623                    }
624                }
625            }
626        }
627        
628        // Compléter avec du silence si nécessaire
629        while samples.len() < frame_size {
630            samples.push(0.0);
631        }
632        
633        Ok(samples)
634    }
635    
636    /// Génère une dissimulation de perte de paquet
637    fn generate_packet_loss_concealment(&self, state: &mut OpusDecoderState) -> Result<Vec<f32>, AppError> {
638        let frame_size = self.config.frame_duration.frame_size(self.config.sample_rate) * self.config.channels as usize;
639        
640        if let Some(ref last_frame) = state.packet_loss_concealment.last_good_frame {
641            // Atténuation progressive basée sur le nombre de pertes consécutives
642            let attenuation = 0.8_f32.powf(state.packet_loss_concealment.consecutive_losses as f32);
643            
644            let mut concealed = Vec::with_capacity(frame_size);
645            for (i, &sample) in last_frame.iter().enumerate() {
646                if concealed.len() < frame_size {
647                    // Atténuation avec légère variation pour éviter les artefacts
648                    let variation = (i as f32 * 0.1).sin() * 0.01;
649                    concealed.push(sample * attenuation + variation);
650                }
651            }
652            
653            // Compléter si nécessaire
654            while concealed.len() < frame_size {
655                concealed.push(0.0);
656            }
657            
658            Ok(concealed)
659        } else {
660            // Pas de frame précédente, générer du silence
661            Ok(vec![0.0; frame_size])
662        }
663    }
664}

666impl AudioDecoder for OpusDecoderImpl {
667    fn decode(&mut self, data: &[u8]) -> Result<DecodedAudio, AppError> {
668        if data.is_empty() {
669            // Perte de paquet
670            self.decode_with_plc(None)
671        } else {
672            self.decode_with_plc(Some(data))
673        }
674    }
675    
676    fn reset(&mut self) -> Result<(), AppError> {
677        self.output_buffer.clear();
678        
679        {
680            let mut state = self.decoder_state.lock();
681            state.total_frames = 0;
682            state.last_decode_time = None;
683            state.packet_loss_concealment.last_good_frame = None;
684            state.packet_loss_concealment.consecutive_losses = 0;
685            state.packet_loss_concealment.concealment_energy = 0.0;
686        }
687        
688        debug!("Décodeur Opus remis à zéro");
689        Ok(())
690    }
691    
692    fn info(&self) -> DecoderInfo {
693        DecoderInfo {
694            codec_name: "Opus".to_string(),
695            version: "1.3.1".to_string(),
696            sample_rate: self.config.sample_rate,
697            channels: self.config.channels,
698            bit_depth: 16,
699            frame_size: self.config.frame_duration.frame_size(self.config.sample_rate) * self.config.channels as usize,
700        }
701    }

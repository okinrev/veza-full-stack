456    pub fn validate(&self) -> Result<(), ConfigError> {
457        // Validation du port
458        if self.port == 0 || self.port > 65535 {
459            return Err(ConfigError::InvalidPort);
460        }
461
462        // Validation du répertoire audio
463        if self.audio_dir.is_empty() {
464            return Err(ConfigError::InvalidAudioDir);
465        }
466
467        // Validation de la clé secrète en production
468        if matches!(self.environment, Environment::Production) {
469            if self.secret_key == "your-secret-key-change-in-production" {
470                return Err(ConfigError::WeakSecretKey);
471            }
472
473            if self.security.jwt_secret.is_none() {
474                return Err(ConfigError::MissingJwtSecret);
475            }
476        }
477
478        // Validation des limites de fichiers
479        if self.max_file_size == 0 {
480            return Err(ConfigError::InvalidFileSize);
481        }
482
483        if self.max_range_size == 0 || self.max_range_size > self.max_file_size {
484            return Err(ConfigError::InvalidRangeSize);
485        }
486
487        // Validation de la base de données
488        if self.database.max_connections == 0 {
489            return Err(ConfigError::InvalidDatabaseConfig);
490        }
491
492        Ok(())
493    }

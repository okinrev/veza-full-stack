255    /// Démarre une session d'upload
256    pub async fn start_upload(
257        &self,
258        user_id: i64,
259        filename: String,
260        file_size: u64,
261        content_type: String,
262    ) -> Result<Uuid, AppError> {
263        // Validation de base
264        self.validate_upload_request(&filename, file_size, &content_type)?;
265        
266        // Vérifier le nombre d'uploads actifs
267        let active_count = self.active_uploads.read().await.len();
268        if active_count >= self.config.max_concurrent_uploads {
269            return Err(AppError::RateLimitExceeded);
270        }
271        
272        let session_id = Uuid::new_v4();
273        let session = UploadSession {
274            id: session_id,
275            user_id,
276            filename: filename.clone(),
277            file_size,
278            content_type,
279            status: UploadStatus::Uploading { bytes_received: 0 },
280            progress: UploadProgress {
281                total_bytes: file_size,
282                uploaded_bytes: 0,
283                processing_progress: 0.0,
284                current_stage: None,
285                estimated_time_remaining: None,
286                upload_speed_bps: 0,
287            },
288            metadata: None,
289            waveform: None,
290            created_at: SystemTime::now(),
291            updated_at: SystemTime::now(),
292        };
293        
294        // Enregistrer la session
295        self.active_uploads.write().await.insert(session_id, session);
296        
297        // Émettre l'événement
298        let _ = self.event_sender.send(UploadEvent::UploadStarted {
299            session_id,
300            user_id,
301            filename,
302        });
303        
304        info!("Session d'upload démarrée: {} pour utilisateur {}", session_id, user_id);
305        Ok(session_id)
306    }

308    /// Reçoit un chunk de données
309    pub async fn receive_chunk(
310        &self,
311        session_id: Uuid,
312        chunk_data: &[u8],
313        chunk_offset: u64,
314    ) -> Result<(), AppError> {
315        let mut sessions = self.active_uploads.write().await;
316        let session = sessions.get_mut(&session_id)
317            .ok_or_else(|| AppError::UploadSessionNotFound { session_id: session_id.to_string() })?;
318        
319        // Vérifier le status
320        match &session.status {
321            UploadStatus::Uploading { .. } => {},
322            _ => return Err(AppError::InvalidUploadState { 
323                current: format!("{:?}", session.status),
324                expected: "Uploading".to_string()
325            }),
326        }
327        
328        // Mettre à jour le progress
329        let new_uploaded = chunk_offset + chunk_data.len() as u64;
330        session.progress.uploaded_bytes = new_uploaded;
331        session.updated_at = SystemTime::now();
332        
333        // Calculer la vitesse d'upload
334        let elapsed = session.updated_at.duration_since(session.created_at).unwrap_or_default();
335        if elapsed.as_secs() > 0 {
336            session.progress.upload_speed_bps = (new_uploaded / elapsed.as_secs()) as u32;
337        }
338        
339        // Émettre l'événement de progress
340        let _ = self.event_sender.send(UploadEvent::UploadProgress {
341            session_id,
342            progress: session.progress.clone(),
343        });
344        
345        // Si upload terminé, démarrer le processing
346        if new_uploaded >= session.file_size {
347            session.status = UploadStatus::Processing { 
348                stage: ProcessingStage::ValidatingFile 
349            };
350            
351            // Démarrer le processing en arrière-plan
352            let self_clone = self.clone();
353            tokio::spawn(async move {
354                if let Err(e) = self_clone.process_uploaded_file(session_id).await {
355                    error!("Erreur processing fichier {}: {:?}", session_id, e);
356                }
357            });
358        } else {
359            session.status = UploadStatus::Uploading { 
360                bytes_received: new_uploaded 
361            };
362        }
363        
364        Ok(())
365    }

367    /// Traite un fichier uploadé
368    async fn process_uploaded_file(&self, session_id: Uuid) -> Result<(), AppError> {
369        // Étape 1: Extraction des métadonnées
370        self.update_processing_stage(session_id, ProcessingStage::ExtractingMetadata).await?;
371        let _metadata = self.extract_metadata(session_id).await?;
372        
373        // Étape 2: Génération de waveform
374        if self.config.enable_waveform_generation {
375            self.update_processing_stage(session_id, ProcessingStage::GeneratingWaveform).await?;
376            let waveform = self.generate_waveform(session_id, &_metadata).await?;
377            self.update_session_waveform(session_id, waveform).await?;
378        }
379        
380        // Étape 3: Stockage final
381        self.update_processing_stage(session_id, ProcessingStage::UploadingToStorage).await?;
382        let stored_file = self.store_file(session_id, &_metadata).await?;
383        
384        // Marquer comme terminé
385        self.complete_upload(session_id, stored_file.id).await?;
386        
387        Ok(())
388    }

390    /// Valide une demande d'upload
391    fn validate_upload_request(
392        &self,
393        filename: &str,
394        file_size: u64,
395        content_type: &str,
396    ) -> Result<(), AppError> {
397        // Vérifier la taille
398        if file_size > self.config.max_file_size {
399            return Err(AppError::ValidationError(format!(
400                "File too large: {} bytes, max: {} bytes", 
401                file_size, 
402                self.config.max_file_size
403            )));
404        }
405        
406        // Vérifier le format
407        if !self.config.allowed_formats.contains(&content_type.to_string()) {
408            return Err(AppError::ValidationError(format!(
409                "Unsupported format: {}, supported: {:?}", 
410                content_type,
411                self.config.allowed_formats
412            )));
413        }
414        
415        // Vérifier l'extension
416        if let Some(extension) = Path::new(filename).extension() {
417            let ext_str = extension.to_string_lossy().to_lowercase();
418            let valid_extensions = ["mp3", "wav", "flac", "aiff", "ogg", "m4a", "mp4"];
419            if !valid_extensions.contains(&ext_str.as_str()) {
420                return Err(AppError::ValidationError(format!(
421                    "Unsupported extension: {}, supported: {:?}", 
422                    ext_str,
423                    valid_extensions
424                )));
425            }
426        }
427        
428        Ok(())
429    }

442            
443            let _ = self.event_sender.send(UploadEvent::ProcessingStarted {
444                session_id,
445                stage,
446            });
447        }
448        Ok(())
449    }

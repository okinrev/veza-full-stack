131impl CreatorDashboard {
132    pub fn new(creator_id: u64) -> Self {
133        Self {
134            analytics: CreatorAnalytics::new(creator_id),
135            monetization: CreatorMonetization::new(),
136            tools: CreatorTools::new(),
137        }
138    }
139    
140    pub async fn get_analytics_summary(&self) -> Result<AnalyticsSummary, AppError> {
141        Ok(AnalyticsSummary {
142            total_plays: self.analytics.total_plays,
143            total_likes: self.analytics.total_likes,
144            follower_count: self.analytics.follower_count,
145            monthly_revenue: self.analytics.monthly_revenue,
146            top_track: self.analytics.top_tracks.first().map(|t| t.title.clone()),
147        })
148    }
149}

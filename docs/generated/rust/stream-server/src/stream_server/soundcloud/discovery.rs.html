700impl DiscoveryEngine {
701    /// Crée un nouveau moteur de découverte
702    pub async fn new(
703        config: DiscoveryConfig,
704        _social_manager: Arc<SocialManager>,
705    ) -> Result<Self, AppError> {
706        let recommendation_engine = Arc::new(RecommendationEngine::new(config.recommendation_config.clone()).await?);
707        let trending_manager = Arc::new(TrendingManager::new(config.trending_config.clone()));
708        let charts_manager = Arc::new(ChartsManager::new(config.charts_config.clone()));
709        let radio_manager = Arc::new(RadioManager::new(config.radio_config.clone()));
710        let engagement_tracker = Arc::new(EngagementTracker::new());
711        
712        Ok(Self {
713            recommendation_engine,
714            trending_manager,
715            charts_manager,
716            radio_manager,
717            engagement_tracker,
718            config,
719        })
720    }
721    
722    /// Obtient des recommandations personnalisées pour un utilisateur
723    pub async fn get_personalized_recommendations(
724        &self,
725        _user_id: i64,
726        count: usize,
727        _seed_tracks: Option<Vec<Uuid>>,
728    ) -> Result<Vec<RecommendationResult>, AppError> {
729        let count = count.min(self.config.max_recommendations_per_request);
730        
731        // Obtenir le profil utilisateur
732        let user_profile = self.recommendation_engine
733            .get_user_profile(_user_id)
734            .await
735            .unwrap_or_else(|| UserListeningProfile::new(_user_id));
736        
737        // Générer les recommandations selon différents algorithmes
738        let mut recommendations = Vec::new();
739        
740        // 60% collaborative filtering
741        let collaborative_count = (count as f32 * 0.6) as usize;
742        let mut collaborative = self.recommendation_engine
743            .get_collaborative_recommendations(_user_id, collaborative_count)
744            .await?;
745        recommendations.append(&mut collaborative);
746        
747        // 30% content-based
748        let content_count = (count as f32 * 0.3) as usize;
749        let mut content_based = self.recommendation_engine
750            .get_content_based_recommendations(_user_id, content_count, _seed_tracks.clone())
751            .await?;
752        recommendations.append(&mut content_based);
753        
754        // 10% trending/social
755        let trending_count = count - recommendations.len();
756        let mut trending = self.get_trending_recommendations(_user_id, trending_count).await?;
757        recommendations.append(&mut trending);
758        
759        // Diversifier et re-scorer
760        let final_recommendations = self.diversify_and_score_recommendations(
761            recommendations,
762            &user_profile,
763            count,
764        ).await?;
765        
766        // Tracker les recommandations pour analytics
767        self.track_recommendations(&final_recommendations, _user_id).await?;
768        
769        Ok(final_recommendations)
770    }

804        &self,
805        _user_id: i64,
806        count: usize,
807    ) -> Result<Vec<RecommendationResult>, AppError> {
808        // Simulation - obtenir les tracks trending qui correspondent au profil utilisateur
809        let trending = self.trending_manager.get_trending(None, None, Some(count * 2)).await?;
810        
811        let mut recommendations = Vec::new();
812        for (i, track) in trending.iter().enumerate().take(count) {
813            recommendations.push(RecommendationResult {
814                track_id: track.track_id,
815                confidence_score: 0.8 - (i as f32 * 0.01), // Score décroissant
816                reason: RecommendationReason::Trending,

818                metadata: None,
819            });

825    /// Diversifie et score les recommandations
826    async fn diversify_and_score_recommendations(
827        &self,
828        mut recommendations: Vec<RecommendationResult>,
829        user_profile: &UserListeningProfile,
830        target_count: usize,
831    ) -> Result<Vec<RecommendationResult>, AppError> {
832        // Supprimer les doublons
833        recommendations.sort_by_key(|r| r.track_id);
834        recommendations.dedup_by_key(|r| r.track_id);
835        
836        // Re-scorer selon le profil utilisateur
837        for rec in &mut recommendations {
838            rec.confidence_score = self.calculate_personalized_score(rec, user_profile).await;
839        }
840        
841        // Trier par score
842        recommendations.sort_by(|a, b| b.confidence_score.partial_cmp(&a.confidence_score).unwrap());
843        
844        // Diversifier par genre
845        let diversified = self.diversify_by_genre(recommendations, target_count).await;
846        
847        Ok(diversified)
848    }

850    /// Calcule un score personnalisé
851    async fn calculate_personalized_score(
852        &self,
853        recommendation: &RecommendationResult,
854        user_profile: &UserListeningProfile,
855    ) -> f32 {
856        // Simulation de scoring personnalisé
857        let mut score = recommendation.confidence_score;
858        
859        // Ajuster selon les préférences de découverte
860        match recommendation.reason {
861            RecommendationReason::Trending => {
862                score *= user_profile.discovery_preferences.popularity_bias;
863            }
864            RecommendationReason::SimilarToLiked => {
865                score *= 1.0 - user_profile.discovery_preferences.familiarity_ratio;
866            }
867            _ => {}
868        }
869        
870        score.clamp(0.0, 1.0)
871    }

1144        {
1145            let user_stations = self.user_stations.read().await;

1156        let station_id = Uuid::new_v4();
1157        let station = RadioStation {
1158            id: station_id,
1159            user_id,
1160            station_type,
1161            name,
1162            description: None,
1163            seed_tracks: Vec::new(),
1164            generated_queue: VecDeque::new(),
1165            played_tracks: Vec::new(),
1166            current_position: 0,
1167            last_updated: SystemTime::now(),
1168            total_listening_time: Duration::ZERO,
1169        };
1170        
1171        // Ajouter la station
1172        {
1173            let mut active_stations = self.active_stations.write().await;
1174            active_stations.insert(station_id, station);
1175        }
1176        
1177        // Ajouter à l'utilisateur
1178        {
1179            let mut user_stations = self.user_stations.write().await;
1180            user_stations.entry(user_id).or_insert_with(Vec::new).push(station_id);
1181        }
1182

1185    }

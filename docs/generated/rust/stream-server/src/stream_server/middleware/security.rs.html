10pub async fn security_headers_middleware(
11    State(_state): State<AppState>,
12    request: Request,
13    next: Next,
14) -> Result<Response, StatusCode> {
15    // Valider la sécurité de la requête
16    validate_request_security(&request)?;
17    
18    // Traiter la requête
19    let mut response = next.run(request).await;
20    
21    // Ajouter les headers de sécurité
22    add_security_headers(&mut response);
23    
24    Ok(response)
25}

27fn validate_request_security(request: &Request) -> Result<(), StatusCode> {
28    let uri = request.uri();
29    let headers = request.headers();
30    let path = uri.path();
31    let query = uri.query().unwrap_or("");
32    
33    // Vérifier les patterns dangereux dans l'URL
34    if contains_dangerous_patterns(path) || contains_dangerous_patterns(query) {
35        warn!(
36            path = %path,
37            query = %query,
38            "Tentative d'attaque par traversée de répertoire détectée"
39        );
40        return Err(StatusCode::BAD_REQUEST);
41    }
42    
43    // Vérifier les tentatives d'injection
44    if contains_injection_patterns(path) || contains_injection_patterns(query) {
45        warn!(
46            path = %path,
47            query = %query,
48            "Tentative d'injection détectée"
49        );
50        return Err(StatusCode::BAD_REQUEST);
51    }
52    
53    // Vérifier la taille des headers
54    for (name, value) in headers.iter() {
55        if value.len() > 8192 {
56            warn!(
57                header = %name,
58                size = value.len(),
59                "Header trop volumineux détecté"
60            );
61            return Err(StatusCode::BAD_REQUEST);
62        }
63    }
64    
65    // Vérifier les headers suspects
66    if let Some(user_agent) = headers.get("user-agent") {
67        if let Ok(ua_str) = user_agent.to_str() {
68            if ua_str.is_empty() || ua_str.len() > 512 {
69                debug!("User-Agent suspect: {}", ua_str);
70            }
71        }
72    }
73    
74    Ok(())
75}

77fn contains_dangerous_patterns(input: &str) -> bool {
78    let dangerous_patterns = [
79        "../", "..\\", "..%2f", "..%5c",
80        "%2e%2e%2f", "%2e%2e%5c",
81        "etc/passwd", "windows/system32",
82        "/proc/", "/sys/",
83        "\\x00", "%00", // Null bytes
84    ];
85    
86    let input_lower = input.to_lowercase();
87    dangerous_patterns.iter().any(|&pattern| input_lower.contains(pattern))
88}

90fn contains_injection_patterns(input: &str) -> bool {
91    let injection_patterns = [
92        // SQL injection
93        "union select", "drop table", "insert into",
94        "delete from", "update set", "create table",
95        "alter table", "truncate", "exec(",
96        
97        // XSS patterns
98        "<script", "javascript:", "onload=",
99        "onerror=", "eval(", "alert(",
100        "document.cookie", "window.location",
101        
102        // Command injection
103        "$(", "`", ";", "|", "&&", "||",
104        "wget", "curl", "nc ", "netcat",
105    ];
106    
107    let input_lower = input.to_lowercase();
108    injection_patterns.iter().any(|&pattern| input_lower.contains(pattern))
109}

111fn add_security_headers(response: &mut Response) {
112    let headers = response.headers_mut();
113    
114    // Empêcher la détection du type MIME
115    headers.insert(
116        HeaderName::from_static("x-content-type-options"),
117        HeaderValue::from_static("nosniff")
118    );
119    
120    // Empêcher l'affichage dans une iframe
121    headers.insert(
122        HeaderName::from_static("x-frame-options"),
123        HeaderValue::from_static("DENY")
124    );
125    
126    // Activer la protection XSS du navigateur
127    headers.insert(
128        HeaderName::from_static("x-xss-protection"),
129        HeaderValue::from_static("1; mode=block")
130    );
131    
132    // Content Security Policy restrictive
133    headers.insert(
134        HeaderName::from_static("content-security-policy"),
135        HeaderValue::from_static("default-src 'none'; media-src 'self'; connect-src 'self'")
136    );
137    
138    // Politique de référent stricte
139    headers.insert(
140        HeaderName::from_static("referrer-policy"),
141        HeaderValue::from_static("strict-origin-when-cross-origin")
142    );
143    
144    // Permissions Policy (anciennement Feature Policy)
145    headers.insert(
146        HeaderName::from_static("permissions-policy"),
147        HeaderValue::from_static("camera=(), microphone=(), geolocation=()")
148    );
149    
150    // HSTS (si HTTPS)
151    // Note: À activer uniquement en HTTPS
152    // headers.insert(
153    //     HeaderName::from_static("strict-transport-security"),
154    //     HeaderValue::from_static("max-age=31536000; includeSubDomains")
155    // );
156}

210    pub async fn start_delivery_workers(&self) {
211        info!("üì¨ D√©marrage de {} workers de notifications", self.delivery_workers);
212        
213        for worker_id in 0..self.delivery_workers {
214            let service = self.clone();
215            tokio::spawn(async move {
216                service.delivery_worker_loop(worker_id).await;
217            });
218        }
219
220        // Worker de nettoyage des anciennes notifications
221        let service = self.clone();
222        tokio::spawn(async move {
223            service.cleanup_worker().await;
224        });
225    }
226
227    async fn delivery_worker_loop(&self, worker_id: usize) {
228        debug!("Worker de notifications {} d√©marr√©", worker_id);
229        
230        loop {
231            let notification = {
232                let mut pending = self.pending_notifications.write().await;
233                pending.pop_front()
234            };
235
236            if let Some(notification) = notification {
237                debug!("Worker {} traite la notification {}", worker_id, notification.id);
238                self.deliver_notification(notification).await;
239            } else {
240                // Pas de notification, attendre un peu
241                tokio::time::sleep(Duration::from_millis(100)).await;
242            }
243        }
244    }
245
246    async fn cleanup_worker(&self) {
247        let mut interval = tokio::time::interval(Duration::from_secs(3600)); // 1 heure
248        
249        loop {
250            interval.tick().await;
251            self.cleanup_expired_notifications().await;
252            self.cleanup_old_history().await;
253            self.update_statistics().await;
254        }

257    pub async fn send_notification(&self, mut notification: Notification) -> Result<String, NotificationError> {
258        // Valider la notification
259        self.validate_notification(&notification)?;
260        
261        // Appliquer les pr√©f√©rences utilisateur
262        notification = self.apply_user_preferences(notification).await?;
263        
264        // V√©rifier les limites de fr√©quence
265        if !self.check_frequency_limits(&notification).await {
266            return Err(NotificationError::FrequencyLimitExceeded);
267        }
268
269        // G√©n√©rer un ID si n√©cessaire
270        if notification.id.is_empty() {
271            notification.id = uuid::Uuid::new_v4().to_string();
272        }
273
274        // Ajouter √† l'historique
275        {
276            let mut history = self.notification_history.write().await;
277            history.insert(notification.id.clone(), notification.clone());
278        }
279
280        // Ajouter √† la queue de livraison
281        {
282            let mut pending = self.pending_notifications.write().await;
283            pending.push_back(notification.clone());
284        }
285
286        // Mettre √† jour les stats
287        {
288            let mut stats = self.stats.write().await;
289            stats.total_sent += 1;
290        }
291
292        info!("üì® Notification {} ajout√©e √† la queue pour l'utilisateur {}", 
293              notification.id, notification.user_id);
294
295        Ok(notification.id)
296    }

347    async fn deliver_notification(&self, mut notification: Notification) {
348        let start_time = SystemTime::now();
349        
350        for channel in &notification.channels.clone() {
351            let delivery_result = match channel {
352                NotificationChannel::WebSocket => self.deliver_websocket(&notification).await,
353                NotificationChannel::Email => self.deliver_email(&notification).await,
354                NotificationChannel::SMS => self.deliver_sms(&notification).await,
355                NotificationChannel::Push => self.deliver_push(&notification).await,
356                NotificationChannel::InApp => self.deliver_in_app(&notification).await,
357                NotificationChannel::Webhook => self.deliver_webhook(&notification).await,
358            };
359
360            match delivery_result {
361                Ok(_) => {
362                    debug!("‚úÖ Notification {} livr√©e via {:?}", notification.id, channel);
363                    self.update_channel_stats(channel, true, start_time).await;
364                }
365                Err(e) => {
366                    error!("‚ùå √âchec de livraison de la notification {} via {:?}: {:?}", 
367                           notification.id, channel, e);
368                    self.update_channel_stats(channel, false, start_time).await;
369                    self.record_delivery_failure(&notification, channel, &e).await;
370                }
371            }
372        }
373
374        notification.delivered = true;
375        notification.delivery_attempts += 1;
376
377        // Mettre √† jour dans l'historique
378        {
379            let mut history = self.notification_history.write().await;
380            history.insert(notification.id.clone(), notification);
381        }
382
383        // Mettre √† jour les stats globales
384        {
385            let mut stats = self.stats.write().await;
386            stats.total_delivered += 1;
387            stats.delivery_rate = stats.total_delivered as f32 / stats.total_sent as f32;
388        }
389    }

445                stats.total_read += 1;
446                stats.read_rate = stats.total_read as f32 / stats.total_delivered as f32;
447            }
448            
449            Ok(())
450        } else {
451            Err(NotificationError::NotificationNotFound(notification_id.to_string()))

455    pub async fn get_user_notifications(
456        &self,
457        user_id: &str,
458        limit: Option<usize>,
459        offset: Option<usize>,
460        unread_only: bool,
461    ) -> Vec<Notification> {
462        let history = self.notification_history.read().await;
463        let limit = limit.unwrap_or(50);
464        let offset = offset.unwrap_or(0);
465        
466        let mut user_notifications: Vec<_> = history.values()
467            .filter(|n| n.user_id == user_id)
468            .filter(|n| !unread_only || !n.read)
469            .cloned()
470            .collect();
471        
472        // Trier par date de cr√©ation (plus r√©cent en premier)
473        user_notifications.sort_by(|a, b| b.created_at.cmp(&a.created_at));
474        
475        user_notifications.into_iter()
476            .skip(offset)
477            .take(limit)
478            .collect()
479    }

522    async fn apply_user_preferences(&self, mut notification: Notification) -> Result<Notification, NotificationError> {
523        if let Some(prefs) = self.get_user_preferences(&notification.user_id).await {
524            // Filtrer les canaux selon les pr√©f√©rences
525            notification.channels.retain(|channel| {
526                *prefs.enabled_channels.get(channel).unwrap_or(&true)
527            });
528
529            // V√©rifier les pr√©f√©rences par type
530            if let Some(type_pref) = prefs.type_preferences.get(&notification.notification_type) {
531                if !type_pref.enabled {
532                    return Err(NotificationError::NotificationDisabled);
533                }
534                
535                if notification.priority < type_pref.priority_threshold {
536                    return Err(NotificationError::PriorityTooLow);
537                }
538
539                // Utiliser les canaux pr√©f√©r√©s si sp√©cifi√©s
540                if !type_pref.channels.is_empty() {
541                    notification.channels = type_pref.channels.clone();
542                }
543            }
544
545            // V√©rifier les heures de silence
546            if let Some(quiet_hours) = &prefs.quiet_hours {
547                if self.is_in_quiet_hours(quiet_hours).await && notification.priority < NotificationPriority::Critical {
548                    return Err(NotificationError::QuietHours);
549                }
550            }
551        }
552
553        if notification.channels.is_empty() {
554            return Err(NotificationError::NoEnabledChannels);
555        }
556
557        Ok(notification)
558    }

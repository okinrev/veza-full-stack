141            encoding_key,
142            decoding_key,
143            validation,
144            revoked_tokens: Arc::new(tokio::sync::RwLock::new(HashMap::new())),

191    pub async fn generate_tokens(&self, user_info: &UserInfo, remember_me: bool) -> Result<(String, String), AuthError> {
192        let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs();
193        let session_id = uuid::Uuid::new_v4().to_string();
194        
195        // Durée d'expiration selon remember_me
196        let expires_in = if remember_me {
197            self.config.security.jwt_expiration.as_secs() * 7 // 7 fois plus long
198        } else {
199            self.config.security.jwt_expiration.as_secs()
200        };
201
202        let claims = Claims {
203            sub: user_info.id,  // ALIGNED WITH i64
204            username: user_info.username.clone(),
205            email: user_info.email.clone(),
206            roles: user_info.roles.clone(),
207            permissions: user_info.permissions.clone(),
208            exp: now + expires_in,
209            iat: now,
210            iss: "stream_server".to_string(),
211            aud: "stream_server".to_string(),
212            session_id: session_id.clone(),
213        };
214
215        let access_token = encode(&Header::default(), &claims, &self.encoding_key)
216            .map_err(|e| AuthError::TokenGenerationError(e.to_string()))?;
217
218        // Refresh token avec une durée plus longue
219        let refresh_claims = Claims {
220            exp: now + (expires_in * 2), // 2x plus long que l'access token
221            ..claims.clone()
222        };
223
224        let refresh_token = encode(&Header::default(), &refresh_claims, &self.encoding_key)
225            .map_err(|e| AuthError::TokenGenerationError(e.to_string()))?;
226
227        Ok((access_token, refresh_token))
228    }

230    pub async fn validate_token(&self, token: &str) -> TokenValidationResult {
231        match decode::<Claims>(token, &self.decoding_key, &self.validation) {
232            Ok(token_data) => {
233                let claims = token_data.claims;
234                
235                // Vérifier si le token est révoqué
236                let revoked_tokens = self.revoked_tokens.read().await;
237                if revoked_tokens.contains_key(&claims.session_id) {
238                    return TokenValidationResult {
239                        valid: false,
240                        claims: None,
241                        error: Some("Token has been revoked".to_string()),
242                    };
243                }
244
245                // Vérifier l'expiration
246                let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs();
247                if claims.exp < now {
248                    return TokenValidationResult {
249                        valid: false,
250                        claims: None,
251                        error: Some("Token has expired".to_string()),
252                    };
253                }
254
255                TokenValidationResult {
256                    valid: true,
257                    claims: Some(claims),
258                    error: None,
259                }
260            }
261            Err(e) => TokenValidationResult {
262                valid: false,
263                claims: None,
264                error: Some(e.to_string()),
265            }
266        }
267    }

269    pub async fn refresh_token(&self, refresh_token: &str) -> Result<(String, String), AuthError> {
270        let validation_result = self.validate_token(refresh_token).await;
271        
272        if !validation_result.valid {
273            return Err(AuthError::InvalidToken(
274                validation_result.error.unwrap_or("Invalid refresh token".to_string())
275            ));
276        }
277
278        let claims = validation_result.claims.unwrap();
279        
280        // Créer un nouveau UserInfo à partir des claims
281        let user_info = UserInfo {
282            id: claims.sub,
283            username: claims.username,
284            email: claims.email,
285            roles: claims.roles,
286            permissions: claims.permissions,
287            subscription_tier: SubscriptionTier::Free, // À déterminer selon la logique métier
288            created_at: claims.iat,
289            last_login: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
290        };
291
292        // Révoquer l'ancien token
293        self.revoke_token(&claims.session_id).await;
294
295        // Générer de nouveaux tokens
296        self.generate_tokens(&user_info, false).await
297    }

299    pub async fn revoke_token(&self, session_id: &str) {
300        let mut revoked_tokens = self.revoked_tokens.write().await;
301        let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs();
302        revoked_tokens.insert(session_id.to_string(), now);
303        
304        // Nettoyer les anciens tokens révoqués (plus de 24h)
305        let cutoff = now - (24 * 3600);
306        revoked_tokens.retain(|_, &mut revocation_time| revocation_time > cutoff);
307    }

321    pub async fn login(&self, request: LoginRequest) -> Result<LoginResponse, AuthError> {
322        let user_info = self.authenticate_user(&request.username, &request.password).await?;
323        let (access_token, refresh_token) = self.generate_tokens(&user_info, request.remember_me.unwrap_or(false)).await?;
324        
325        Ok(LoginResponse {
326            access_token,
327            refresh_token,
328            token_type: "Bearer".to_string(),
329            expires_in: self.config.security.jwt_expiration.as_secs(),
330            user_info,
331        })

347// Middleware d'authentification
348pub async fn auth_middleware(
349    State(auth_manager): State<Arc<AuthManager>>,
350    mut request: Request,
351    next: Next,
352) -> Result<Response, StatusCode> {
353    let headers = request.headers();
354    
355    let token = extract_token_from_headers(headers)
356        .ok_or(StatusCode::UNAUTHORIZED)?;
357
358    let validation_result = auth_manager.validate_token(&token).await;
359    
360    if !validation_result.valid {
361        warn!("Token validation failed: {:?}", validation_result.error);
362        return Err(StatusCode::UNAUTHORIZED);
363    }
364
365    let claims = validation_result.claims.unwrap();
366    
367    // Ajouter les claims à la requête pour les handlers suivants
368    request.extensions_mut().insert(claims);
369    
370    Ok(next.run(request).await)
371}

373// Middleware de vérification des permissions
374pub fn require_permission(required_permission: Permission) -> impl Fn(Request, Next) -> std::pin::Pin<Box<dyn std::future::Future<Output = Result<Response, StatusCode>> + Send>> + Clone {
375    move |request: Request, next: Next| {
376        let required_permission = required_permission.clone();
377        Box::pin(async move {
378            let claims = request.extensions().get::<Claims>()
379                .ok_or(StatusCode::UNAUTHORIZED)?;
380
381            if !claims.permissions.contains(&required_permission) {
382                warn!("User {} lacks required permission: {:?}", claims.username, required_permission);
383                return Err(StatusCode::FORBIDDEN);
384            }
385
386            Ok(next.run(request).await)
387        })
388    }
389}

409fn extract_token_from_headers(headers: &HeaderMap) -> Option<String> {
410    let auth_header = headers.get("Authorization")?;
411    let auth_str = auth_header.to_str().ok()?;
412    
413    if auth_str.starts_with("Bearer ") {
414        Some(auth_str[7..].to_string())
415    } else {
416        None
417    }
418}

421pub async fn login_handler(
422    State(auth_manager): State<Arc<AuthManager>>,
423    Json(request): Json<LoginRequest>,
424) -> Result<Json<LoginResponse>, (StatusCode, String)> {
425    match auth_manager.login(request).await {
426        Ok(response) => Ok(Json(response)),
427        Err(e) => {
428            error!("Login failed: {:?}", e);
429            match e {
430                AuthError::InvalidCredentials => Err((StatusCode::UNAUTHORIZED, "Invalid credentials".to_string())),
431                AuthError::TokenGenerationError(msg) => Err((StatusCode::INTERNAL_SERVER_ERROR, msg)),
432                _ => Err((StatusCode::INTERNAL_SERVER_ERROR, "Authentication failed".to_string())),
433            }
434        }
435    }
436}

438pub async fn refresh_handler(
439    State(auth_manager): State<Arc<AuthManager>>,
440    Json(request): Json<RefreshTokenRequest>,
441) -> Result<Json<LoginResponse>, (StatusCode, String)> {
442    match auth_manager.refresh_token(&request.refresh_token).await {
443        Ok((access_token, refresh_token)) => {
444            // Valider le refresh token pour récupérer les claims
445            let validation_result = auth_manager.validate_token(&request.refresh_token).await;
446            let claims = validation_result.claims.unwrap(); // Safe because refresh_token succeeded
447            
448            // Créer UserInfo à partir des claims avec i64 aligned
449            let user_info = UserInfo {
450                id: claims.sub,  // ALIGNED WITH i64 - use claims.sub directly
451                username: claims.username.clone(),
452                email: claims.email.clone(),
453                roles: claims.roles.clone(),
454                permissions: claims.permissions.clone(),
455                subscription_tier: SubscriptionTier::Free,
456                created_at: claims.iat,
457                last_login: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
458            };
459
460            let response = LoginResponse {
461                access_token,
462                refresh_token,
463                token_type: "Bearer".to_string(),
464                expires_in: auth_manager.config.security.jwt_expiration.as_secs(),
465                user_info,
466            };
467
468            Ok(Json(response))
469        }
470        Err(e) => {
471            error!("Token refresh failed: {:?}", e);
472            Err((StatusCode::UNAUTHORIZED, "Token refresh failed".to_string()))
473        }
474    }

477pub async fn logout_handler(
478    State(auth_manager): State<Arc<AuthManager>>,
479    request: Request,
480) -> Result<Json<serde_json::Value>, (StatusCode, String)> {
481    if let Some(claims) = request.extensions().get::<Claims>() {
482        auth_manager.revoke_token(&claims.session_id).await;
483        debug!("User {} logged out", claims.username);
484    }
485
486    Ok(Json(serde_json::json!({
487        "message": "Successfully logged out"
488    })))
489}

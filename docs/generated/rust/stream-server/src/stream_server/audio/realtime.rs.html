13/// Buffer circulaire thread-safe pour audio temps réel
14#[derive(Debug)]
15pub struct RingBuffer<T> {
16    buffer: Vec<T>,
17    write_index: usize,
18    read_index: usize,
19    size: usize,
20    filled: bool,
21}
22
23impl<T: Clone + Default> RingBuffer<T> {
24    pub fn new(size: usize) -> Self {
25        Self {
26            buffer: vec![T::default(); size],
27            write_index: 0,
28            read_index: 0,
29            size,
30            filled: false,
31        }
32    }
33    
34    pub fn write(&mut self, data: &[T]) -> usize {
35        let mut written = 0;
36        
37        for item in data.iter() {
38            if self.is_full() && !self.filled {
39                break;
40            }
41            
42            self.buffer[self.write_index] = item.clone();
43            self.write_index = (self.write_index + 1) % self.size;
44            written += 1;
45            
46            if self.write_index == self.read_index {
47                self.filled = true;
48            }
49        }
50        
51        written
52    }
53    
54    pub fn read(&mut self, data: &mut [T]) -> usize {
55        let mut read = 0;
56        
57        for item in data.iter_mut() {
58            if self.is_empty() {
59                break;
60            }
61            
62            *item = self.buffer[self.read_index].clone();
63            self.read_index = (self.read_index + 1) % self.size;
64            read += 1;
65            
66            if self.read_index == self.write_index {
67                self.filled = false;
68            }
69        }
70        
71        read
72    }
73    
74    pub fn is_empty(&self) -> bool {
75        !self.filled && self.read_index == self.write_index
76    }
77    
78    pub fn is_full(&self) -> bool {
79        self.filled
80    }
81    
82    pub fn available_read(&self) -> usize {
83        if self.is_empty() {
84            0
85        } else if self.filled {
86            self.size
87        } else if self.write_index >= self.read_index {
88            self.write_index - self.read_index
89        } else {
90            self.size - self.read_index + self.write_index
91        }
92    }
93    
94    pub fn available_write(&self) -> usize {
95        self.size - self.available_read()
96    }
97}

219impl RealtimeAudioProcessor {
220    pub fn new(config: RealtimeConfig) -> Result<Self, AppError> {
221        let buffer_size = config.buffer_size;
222        
223        Ok(Self {
224            input_buffer: Arc::new(Mutex::new(RingBuffer::new(buffer_size * 4))),
225            output_buffer: Arc::new(Mutex::new(RingBuffer::new(buffer_size * 4))),
226            effects_chain: Arc::new(Mutex::new(EffectsChain::new())),
227            config,
228            metrics: Arc::new(RwLock::new(RealtimeMetrics::default())),
229            processing_thread: None,
230            shutdown_signal: Arc::new(std::sync::atomic::AtomicBool::new(false)),
231        })
232    }
233    
234    /// Démarre le traitement temps réel
235    pub fn start(&mut self) -> Result<(), AppError> {
236        if self.processing_thread.is_some() {
237            return Err(AppError::AlreadyRunning);
238        }
239        
240        self.shutdown_signal.store(false, std::sync::atomic::Ordering::Relaxed);
241        
242        let input_buffer = self.input_buffer.clone();
243        let output_buffer = self.output_buffer.clone();
244        let effects_chain = self.effects_chain.clone();
245        let config = self.config.clone();
246        let metrics = self.metrics.clone();
247        let shutdown = self.shutdown_signal.clone();
248        
249        let handle = std::thread::Builder::new()
250            .name("realtime-audio".to_string())
251            .spawn(move || {
252                Self::processing_loop(
253                    input_buffer,
254                    output_buffer,
255                    effects_chain,
256                    config,
257                    metrics,
258                    shutdown,
259                );
260            })
261            .map_err(|e| AppError::ThreadError { 
262                message: format!("Failed to start audio thread: {}", e) 
263            })?;
264        
265        self.processing_thread = Some(handle);
266        Ok(())
267    }
268    
269    /// Arrête le traitement
270    pub fn stop(&mut self) -> Result<(), AppError> {
271        if let Some(handle) = self.processing_thread.take() {
272            self.shutdown_signal.store(true, std::sync::atomic::Ordering::Relaxed);
273            handle.join().map_err(|_| AppError::ThreadError { 
274                message: "Failed to join audio thread".to_string() 
275            })?;
276        }
277        Ok(())
278    }
279    
280    /// Boucle de traitement principale
281    fn processing_loop(
282        input_buffer: Arc<Mutex<RingBuffer<f32>>>,
283        output_buffer: Arc<Mutex<RingBuffer<f32>>>,
284        effects_chain: Arc<Mutex<EffectsChain>>,
285        config: RealtimeConfig,
286        metrics: Arc<RwLock<RealtimeMetrics>>,
287        shutdown: Arc<std::sync::atomic::AtomicBool>,
288    ) {
289        let mut processing_buffer = vec![0.0f32; config.buffer_size];
290        let frame_duration = Duration::from_micros(
291            (config.buffer_size as u64 * 1_000_000) / config.sample_rate as u64
292        );
293        
294        while !shutdown.load(std::sync::atomic::Ordering::Relaxed) {
295            let start_time = Instant::now();
296            
297            // Lecture depuis le buffer d'entrée
298            let samples_read = {
299                let mut input = input_buffer.lock();
300                input.read(&mut processing_buffer)
301            };
302            
303            if samples_read == 0 {
304                // Buffer underrun
305                let mut metrics_guard = metrics.write();
306                metrics_guard.buffer_underruns += 1;
307                
308                // Attendre un peu avant de réessayer
309                std::thread::sleep(Duration::from_micros(100));
310                continue;
311            }
312            
313            // Traitement des effets
314            {
315                let mut effects = effects_chain.lock();
316                if let Err(e) = effects.process(
317                    &mut processing_buffer[..samples_read],
318                    config.sample_rate,
319                    config.channels,
320                ) {
321                    eprintln!("Effect processing error: {:?}", e);
322                }
323            }
324            
325            // Écriture vers le buffer de sortie
326            let samples_written = {
327                let mut output = output_buffer.lock();
328                output.write(&processing_buffer[..samples_read])
329            };
330            
331            if samples_written < samples_read {
332                // Buffer overrun
333                let mut metrics_guard = metrics.write();
334                metrics_guard.buffer_overruns += 1;
335            }
336            
337            // Mise à jour des métriques
338            let processing_time = start_time.elapsed();
339            let mut metrics_guard = metrics.write();
340            metrics_guard.current_latency_us = processing_time.as_micros() as u64;
341            metrics_guard.samples_processed += samples_read as u64;
342            
343            // Calcul utilisation CPU (approximation)
344            let cpu_usage = (processing_time.as_micros() as f32 / frame_duration.as_micros() as f32) * 100.0;
345            metrics_guard.cpu_usage_percent = cpu_usage.min(100.0);
346            
347            // Attendre pour maintenir le timing
348            if processing_time < frame_duration {
349                std::thread::sleep(frame_duration - processing_time);
350            }
351        }
352    }
353    
354    /// Ajoute des données audio à traiter
355    pub fn write_input(&self, samples: &[f32]) -> Result<usize, AppError> {
356        let mut input = self.input_buffer.lock();
357        Ok(input.write(samples))
358    }
359    
360    /// Lit les données audio traitées
361    pub fn read_output(&self, buffer: &mut [f32]) -> Result<usize, AppError> {
362        let mut output = self.output_buffer.lock();
363        Ok(output.read(buffer))
364    }
365    
366    /// Obtient les métriques actuelles
367    pub fn get_metrics(&self) -> RealtimeMetrics {
368        self.metrics.read().clone()
369    }
370    
371    /// Ajoute un effet à la chaîne
372    pub fn add_effect(&self, effect: Box<dyn AudioEffect>) {
373        let mut effects = self.effects_chain.lock();
374        effects.add_effect(effect);
375    }

378impl AdaptiveResampler {
379    pub fn new(config: ResamplerConfig) -> Self {
380        Self {
381            ratio: 1.0,
382            target_ratio: 1.0,
383            _filter_coeffs: Self::generate_filter_coeffs(&config),
384            _history: VecDeque::with_capacity(64),
385            config,
386        }
387    }
388    
389    fn generate_filter_coeffs(config: &ResamplerConfig) -> Vec<f32> {
390        match config.filter_quality {
391            FilterQuality::Low => vec![1.0], // Pas de filtrage
392            FilterQuality::Medium => vec![0.25, 0.5, 0.25], // Filtre simple
393            FilterQuality::High => {
394                // Filtre Kaiser-Bessel
395                let taps = 32;
396                let mut coeffs = Vec::with_capacity(taps);
397                for i in 0..taps {
398                    let x = (i as f32 - taps as f32 / 2.0) / (taps as f32 / 2.0);
399                    let sinc = if x == 0.0 { 1.0 } else { (std::f32::consts::PI * x).sin() / (std::f32::consts::PI * x) };
400                    let window = 0.54 - 0.46 * (2.0 * std::f32::consts::PI * i as f32 / (taps - 1) as f32).cos();
401                    coeffs.push(sinc * window);
402                }
403                coeffs
404            },
405            FilterQuality::Audiophile => {
406                // Filtre très haute qualité (plus de taps)
407                let taps = 128;
408                let mut coeffs = Vec::with_capacity(taps);
409                for i in 0..taps {
410                    let x = (i as f32 - taps as f32 / 2.0) / (taps as f32 / 2.0);
411                    let sinc = if x == 0.0 { 1.0 } else { (std::f32::consts::PI * x).sin() / (std::f32::consts::PI * x) };
412                    let window = 0.54 - 0.46 * (2.0 * std::f32::consts::PI * i as f32 / (taps - 1) as f32).cos();
413                    coeffs.push(sinc * window);
414                }
415                coeffs
416            }
417        }
418    }
419    
420    /// Resample un buffer audio
421    pub fn process(&mut self, input: &[f32], output: &mut Vec<f32>) -> Result<(), AppError> {
422        // Mise à jour progressive du ratio vers la cible
423        let ratio_diff = self.target_ratio - self.ratio;
424        self.ratio += ratio_diff * self.config.adaptation_speed;
425        
426        output.clear();
427        
428        // Resampling avec interpolation linéaire (simplifié)
429        let mut input_pos = 0.0;
430        
431        while input_pos < input.len() as f64 - 1.0 {
432            let index = input_pos as usize;
433            let frac = input_pos - index as f64;
434            
435            // Interpolation linéaire
436            let sample = if index + 1 < input.len() {
437                input[index] * (1.0 - frac as f32) + input[index + 1] * frac as f32
438            } else {
439                input[index]
440            };
441            
442            output.push(sample);
443            input_pos += self.ratio;
444        }
445        
446        Ok(())
447    }
448    
449    /// Ajuste le ratio de resampling
450    pub fn set_target_ratio(&mut self, ratio: f64) {
451        let max_dev = self.config.max_ratio_deviation;
452        self.target_ratio = ratio.clamp(1.0 - max_dev, 1.0 + max_dev);
453    }
454}

456impl PIDController {
457    pub fn new(kp: f32, ki: f32, kd: f32) -> Self {
458        Self {
459            kp, ki, kd,
460            integral: 0.0,
461            previous_error: 0.0,
462            last_update: Instant::now(),
463        }
464    }
465    
466    pub fn update(&mut self, setpoint: f32, measured: f32) -> f32 {
467        let now = Instant::now();
468        let dt = now.duration_since(self.last_update).as_secs_f32();
469        self.last_update = now;
470        
471        let error = setpoint - measured;
472        
473        // Terme proportionnel
474        let p_term = self.kp * error;
475        
476        // Terme intégral
477        self.integral += error * dt;
478        let i_term = self.ki * self.integral;
479        
480        // Terme dérivé
481        let derivative = (error - self.previous_error) / dt;
482        let d_term = self.kd * derivative;
483        
484        self.previous_error = error;
485        
486        p_term + i_term + d_term
487    }
488}

521        let mut buffer = RingBuffer::new(4);
522        
523        // Test écriture

525        assert_eq!(written, 3);

527        
528        // Test lecture
529        let mut output = [0.0; 2];
530        let read = buffer.read(&mut output);
531        assert_eq!(read, 2);
532        assert_eq!(output, [1.0, 2.0]);
533        assert_eq!(buffer.available_read(), 1);

1/// Gestion centralisée des erreurs du stream server

5use std::fmt;
6use axum::http::StatusCode;
7use axum::response::{IntoResponse, Response};
8use axum::Json;
9use serde_json::json;
10use serde::{Serialize, Deserialize};

12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub enum AppError {
14    // Configuration et initialisation
15    ConfigError { message: String },
16    Config(String), // Variante pour compatibilité
17    InitializationError { message: String },
18    
19    // Erreurs réseau et connexion
20    NetworkError { message: String },
21    ConnectionClosed,
22    ConnectionTimeout,
23    
24    // Erreurs audio et streaming
25    AudioError { message: String },
26    StreamingError { message: String },
27    EncodingError { message: String },
28    DecodingError { message: String },
29    
30    // Erreurs de codec
31    UnsupportedCodec { codec: String },
32    InvalidSampleRate { rate: u32 },
33    InvalidChannelCount { channels: u8 },
34    InvalidBitrate { bitrate: u32, codec: String },
35    
36    // Erreurs de validation et parsing
37    ValidationError(String),
38    ParseError(String),
39    ParameterMismatch { expected: String, got: String },
40    InvalidRange,
41    
42    // Erreurs de synchronisation
43    TimeSync,
44    NoSyncPoint,
45    
46    // Erreurs de playback
47    InvalidPlaybackState { state: String },
48    RateLimitExceeded,
49    LimitExceeded { resource: String, limit: u32 },
50    ListenerLimitExceeded { current: u32, limit: u32 },
51    UploadSessionNotFound { session_id: String },
52    InvalidUploadState { current: String, expected: String },
53    TooManyActivePlayers { limit: u32 },
54    PlayerNotFound { user_id: i64 },
55    
56    // Erreurs de fichier et stockage
57    FileError { message: String },
58    FileNotFound,
59    NotFound { resource: String },
60    StorageError { message: String },
61    
62    // Erreurs de sérialisation
63    SerializationError,
64    InvalidData { message: String },
65    
66    // Erreurs de ressources
67    ResourceExhausted { resource: String },
68    NotEnoughData,
69    BufferOverflow,
70    
71    // Erreurs de buffer
72    BufferNotFound { stream_id: String },
73    BufferFull { stream_id: String },
74    InsufficientData,
75    
76    // Erreurs d'autorisation
77    Unauthorized,
78    Forbidden,
79    
80    // Erreurs de thread et concurrence
81    AlreadyRunning,
82    AlreadyProcessing,
83    ThreadError { message: String },
84    
85    // Erreurs génériques
86    InternalError { message: String },
87    ExternalServiceError { service: String, message: String },
88}

90impl fmt::Display for AppError {

92        match self {
93            AppError::ConfigError { message } => write!(f, "Configuration error: {}", message),
94            AppError::Config(message) => write!(f, "Configuration error: {}", message),
95            AppError::InitializationError { message } => write!(f, "Initialization error: {}", message),
96            AppError::NetworkError { message } => write!(f, "Network error: {}", message),
97            AppError::ConnectionClosed => write!(f, "Connection closed"),
98            AppError::ConnectionTimeout => write!(f, "Connection timeout"),
99            AppError::AudioError { message } => write!(f, "Audio error: {}", message),
100            AppError::StreamingError { message } => write!(f, "Streaming error: {}", message),
101            AppError::EncodingError { message } => write!(f, "Encoding error: {}", message),
102            AppError::DecodingError { message } => write!(f, "Decoding error: {}", message),
103            AppError::UnsupportedCodec { codec } => write!(f, "Unsupported codec: {}", codec),
104            AppError::InvalidSampleRate { rate } => write!(f, "Invalid sample rate: {}", rate),
105            AppError::InvalidChannelCount { channels } => write!(f, "Invalid channel count: {}", channels),
106            AppError::InvalidBitrate { bitrate, codec } => write!(f, "Invalid bitrate: {} for codec: {}", bitrate, codec),
107            AppError::ValidationError(message) => write!(f, "Validation error: {}", message),
108            AppError::ParseError(message) => write!(f, "Parse error: {}", message),
109            AppError::ParameterMismatch { expected, got } => write!(f, "Parameter mismatch: expected {} but got {}", expected, got),
110            AppError::InvalidRange => write!(f, "Invalid range request"),
111            AppError::TimeSync => write!(f, "Time synchronization error"),
112            AppError::NoSyncPoint => write!(f, "No synchronization point found"),
113            AppError::InvalidPlaybackState { state } => write!(f, "Invalid playback state: {}", state),
114            AppError::RateLimitExceeded => write!(f, "Rate limit exceeded"),
115            AppError::LimitExceeded { resource, limit } => write!(f, "Limit exceeded for {}: max {}", resource, limit),
116            AppError::ListenerLimitExceeded { current, limit } => write!(f, "Listener limit exceeded: {} current, {} limit", current, limit),
117            AppError::UploadSessionNotFound { session_id } => write!(f, "Upload session not found: {}", session_id),
118            AppError::InvalidUploadState { current, expected } => write!(f, "Invalid upload state: current {} but expected {}", current, expected),
119            AppError::TooManyActivePlayers { limit } => write!(f, "Too many active players: limit {}", limit),
120            AppError::PlayerNotFound { user_id } => write!(f, "Player not found: user_id {}", user_id),
121            AppError::FileError { message } => write!(f, "File error: {}", message),
122            AppError::FileNotFound => write!(f, "File not found"),
123            AppError::NotFound { resource } => write!(f, "Not found: {}", resource),
124            AppError::StorageError { message } => write!(f, "Storage error: {}", message),
125            AppError::SerializationError => write!(f, "Serialization error"),
126            AppError::InvalidData { message } => write!(f, "Invalid data: {}", message),
127            AppError::ResourceExhausted { resource } => write!(f, "Resource exhausted: {}", resource),
128            AppError::NotEnoughData => write!(f, "Not enough data"),
129            AppError::BufferOverflow => write!(f, "Buffer overflow"),
130            AppError::BufferNotFound { stream_id } => write!(f, "Buffer not found for stream: {}", stream_id),
131            AppError::BufferFull { stream_id } => write!(f, "Buffer full for stream: {}", stream_id),
132            AppError::InsufficientData => write!(f, "Insufficient data"),
133            AppError::Unauthorized => write!(f, "Unauthorized access"),
134            AppError::Forbidden => write!(f, "Forbidden access"),
135            AppError::AlreadyRunning => write!(f, "Process already running"),
136            AppError::AlreadyProcessing => write!(f, "Already processing"),
137            AppError::ThreadError { message } => write!(f, "Thread error: {}", message),
138            AppError::InternalError { message } => write!(f, "Internal error: {}", message),
139            AppError::ExternalServiceError { service, message } => write!(f, "External service error: {} - {}", service, message),
140        }
141    }
142}

146impl IntoResponse for AppError {
147    fn into_response(self) -> Response {
148        let (status, error_message) = match self {
149            AppError::ConfigError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
150            AppError::Config(message) => (StatusCode::INTERNAL_SERVER_ERROR, message),
151            AppError::InitializationError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
152            AppError::NetworkError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
153            AppError::ConnectionClosed => (StatusCode::INTERNAL_SERVER_ERROR, "Connection closed".to_string()),
154            AppError::ConnectionTimeout => (StatusCode::INTERNAL_SERVER_ERROR, "Connection timeout".to_string()),
155            AppError::AudioError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
156            AppError::StreamingError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
157            AppError::EncodingError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
158            AppError::DecodingError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
159            AppError::UnsupportedCodec { codec } => (StatusCode::BAD_REQUEST, format!("Unsupported codec: {}", codec)),
160            AppError::InvalidSampleRate { rate } => (StatusCode::BAD_REQUEST, format!("Invalid sample rate: {}", rate)),
161            AppError::InvalidChannelCount { channels } => (StatusCode::BAD_REQUEST, format!("Invalid channel count: {}", channels)),
162            AppError::InvalidBitrate { bitrate, codec } => (StatusCode::BAD_REQUEST, format!("Invalid bitrate: {} for codec: {}", bitrate, codec)),
163            AppError::ValidationError(message) => (StatusCode::BAD_REQUEST, message),
164            AppError::ParseError(message) => (StatusCode::BAD_REQUEST, message),
165            AppError::ParameterMismatch { expected, got } => (StatusCode::BAD_REQUEST, format!("Parameter mismatch: expected {} but got {}", expected, got)),
166            AppError::InvalidRange => (StatusCode::RANGE_NOT_SATISFIABLE, "Invalid range request".to_string()),
167            AppError::TimeSync => (StatusCode::INTERNAL_SERVER_ERROR, "Time synchronization error".to_string()),
168            AppError::NoSyncPoint => (StatusCode::INTERNAL_SERVER_ERROR, "No synchronization point found".to_string()),
169            AppError::InvalidPlaybackState { state } => (StatusCode::BAD_REQUEST, format!("Invalid playback state: {}", state)),
170            AppError::RateLimitExceeded => (StatusCode::TOO_MANY_REQUESTS, "Rate limit exceeded".to_string()),
171            AppError::LimitExceeded { resource, limit } => (StatusCode::TOO_MANY_REQUESTS, format!("Limit exceeded for {}: max {}", resource, limit)),
172            AppError::ListenerLimitExceeded { current, limit } => (StatusCode::TOO_MANY_REQUESTS, format!("Listener limit exceeded: {} current, {} limit", current, limit)),
173            AppError::UploadSessionNotFound { session_id } => (StatusCode::NOT_FOUND, format!("Upload session not found: {}", session_id)),
174            AppError::InvalidUploadState { current, expected } => (StatusCode::BAD_REQUEST, format!("Invalid upload state: current {} but expected {}", current, expected)),
175            AppError::TooManyActivePlayers { limit } => (StatusCode::BAD_REQUEST, format!("Too many active players: limit {}", limit)),
176            AppError::PlayerNotFound { user_id } => (StatusCode::NOT_FOUND, format!("Player not found: user_id {}", user_id)),
177            AppError::FileError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
178            AppError::FileNotFound => (StatusCode::NOT_FOUND, "File not found".to_string()),
179            AppError::NotFound { resource } => (StatusCode::NOT_FOUND, format!("Not found: {}", resource)),
180            AppError::StorageError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
181            AppError::SerializationError => (StatusCode::INTERNAL_SERVER_ERROR, "Serialization error".to_string()),
182            AppError::InvalidData { message } => (StatusCode::BAD_REQUEST, message),
183            AppError::ResourceExhausted { resource } => (StatusCode::INTERNAL_SERVER_ERROR, format!("Resource exhausted: {}", resource)),
184            AppError::NotEnoughData => (StatusCode::BAD_REQUEST, "Not enough data".to_string()),
185            AppError::BufferOverflow => (StatusCode::BAD_REQUEST, "Buffer overflow".to_string()),
186            AppError::BufferNotFound { stream_id } => (StatusCode::NOT_FOUND, format!("Buffer not found for stream: {}", stream_id)),
187            AppError::BufferFull { stream_id } => (StatusCode::TOO_MANY_REQUESTS, format!("Buffer full for stream: {}", stream_id)),
188            AppError::InsufficientData => (StatusCode::BAD_REQUEST, "Insufficient data".to_string()),
189            AppError::Unauthorized => (StatusCode::UNAUTHORIZED, "Unauthorized access".to_string()),
190            AppError::Forbidden => (StatusCode::FORBIDDEN, "Forbidden access".to_string()),
191            AppError::AlreadyRunning => (StatusCode::CONFLICT, "Process already running".to_string()),
192            AppError::AlreadyProcessing => (StatusCode::CONFLICT, "Already processing".to_string()),
193            AppError::ThreadError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
194            AppError::InternalError { message } => (StatusCode::INTERNAL_SERVER_ERROR, message),
195            AppError::ExternalServiceError { service, message } => (StatusCode::INTERNAL_SERVER_ERROR, format!("External service error: {} - {}", service, message)),
196        };
197
198        let body = Json(json!({
199            "error": error_message,
200            "status": status.as_u16(),
201        }));
202
203        (status, body).into_response()
204    }
205}

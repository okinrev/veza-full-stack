215    pub async fn start_play_session(
216        &self,
217        user_id: Option<String>,
218        track_id: String,
219        client_ip: String,
220        user_agent: Option<String>,
221        total_duration_ms: u64,
222        quality: String,
223        platform: Platform,
224        referrer: Option<String>,
225    ) -> Uuid {
226        let session_id = Uuid::new_v4();
227        let now = SystemTime::now();
228
229        let session = PlaySession {
230            session_id,
231            user_id: user_id.clone(),
232            track_id: track_id.clone(),
233            client_ip: client_ip.clone(),
234            user_agent,
235            started_at: now,
236            last_update: now,
237            duration_played_ms: 0,
238            total_duration_ms,
239            completion_percentage: 0.0,
240            quality: quality.clone(),
241            platform: platform.clone(),
242            location: None, // TODO: Géolocalisation IP
243            referrer,
244            ended: false,
245            skip_reason: None,
246        };
247
248        // Sauvegarder en base
249        let started_at_ts = now.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64;
250        
251        if let Err(e) = sqlx::query(r#"
252            INSERT INTO play_sessions (
253                session_id, user_id, track_id, client_ip, user_agent, started_at, 
254                last_update, duration_played_ms, total_duration_ms, completion_percentage,
255                quality, platform, ended
256            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
257        "#)
258        .bind(session_id.to_string())
259        .bind(&user_id)
260        .bind(&track_id)
261        .bind(&client_ip)
262        .bind(&session.user_agent)
263        .bind(started_at_ts)
264        .bind(started_at_ts)
265        .bind(0i64)
266        .bind(total_duration_ms as i64)
267        .bind(0.0f64)
268        .bind(&quality)
269        .bind(format!("{:?}", platform))
270        .bind(false)
271        .execute(&self.db_pool).await {
272            error!("Erreur sauvegarde session: {}", e);
273        }
274
275        // Mettre en cache
276        {
277            let mut sessions = self.active_sessions.write().await;
278            sessions.insert(session_id, session);
279        }
280
281        // Mettre à jour les stats temps réel
282        {
283            let mut stats = self.realtime_stats.write().await;
284            stats.current_listeners += 1;
285            stats.streams_started_last_hour += 1;
286            *stats.quality_distribution_now.entry(quality).or_insert(0) += 1;
287        }
288
289        debug!("Session de lecture démarrée: {} pour track: {}", session_id, track_id);
290        session_id
291    }

293    /// Met à jour le progrès d'une session
294    pub async fn update_play_progress(
295        &self,
296        session_id: Uuid,
297        duration_played_ms: u64,
298        buffer_health: Option<f32>,
299    ) {
300        let now = SystemTime::now();
301        
302        // Mettre à jour en mémoire
303        {
304            let mut sessions = self.active_sessions.write().await;
305            if let Some(session) = sessions.get_mut(&session_id) {
306                session.duration_played_ms = duration_played_ms;
307                session.last_update = now;
308                session.completion_percentage = if session.total_duration_ms > 0 {
309                    (duration_played_ms as f32 / session.total_duration_ms as f32) * 100.0
310                } else {
311                    0.0
312                };
313            }
314        }
315
316        // Mettre à jour périodiquement en base (toutes les 30 secondes)
317        let last_update_ts = now.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64;
318        
319        if let Err(e) = sqlx::query(r#"
320            UPDATE play_sessions 
321            SET last_update = ?, duration_played_ms = ?, completion_percentage = ?
322            WHERE session_id = ?
323        "#)
324        .bind(last_update_ts)
325        .bind(duration_played_ms as i64)
326        .bind((duration_played_ms as f32 / 100.0).min(100.0))
327        .bind(session_id.to_string())
328        .execute(&self.db_pool).await {
329            debug!("Erreur mise à jour session: {}", e);
330        }
331
332        // Mettre à jour buffer health global
333        if let Some(health) = buffer_health {
334            let mut stats = self.realtime_stats.write().await;
335            stats.average_buffer_health = (stats.average_buffer_health * 0.9) + (health * 0.1);
336        }
337    }

339    /// Termine une session de lecture
340    pub async fn end_play_session(&self, session_id: Uuid, skip_reason: Option<SkipReason>) {
341        let session = {
342            let mut sessions = self.active_sessions.write().await;
343            sessions.remove(&session_id)
344        };
345
346        if let Some(mut session) = session {
347            session.ended = true;
348            session.skip_reason = skip_reason.clone();
349            session.last_update = SystemTime::now();
350
351            // Sauvegarder en base
352            let last_update_ts = session.last_update.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64;
353            
354            if let Err(e) = sqlx::query(r#"
355                UPDATE play_sessions 
356                SET last_update = ?, duration_played_ms = ?, completion_percentage = ?, 
357                    ended = ?, skip_reason = ?
358                WHERE session_id = ?
359            "#)
360            .bind(last_update_ts)
361            .bind(session.duration_played_ms as i64)
362            .bind(session.completion_percentage as f64)
363            .bind(true)
364            .bind(skip_reason.as_ref().map(|r| format!("{:?}", r)))
365            .bind(session_id.to_string())
366            .execute(&self.db_pool).await {
367                error!("Erreur fin de session: {}", e);
368            }
369
370            // Mettre à jour les analytics
371            self.update_track_analytics(&session.track_id, &session).await;
372            if let Some(ref user_id) = session.user_id {
373                self.update_user_analytics(user_id, &session).await;
374            }
375
376            // Mettre à jour les stats temps réel
377            {
378                let mut stats = self.realtime_stats.write().await;
379                stats.current_listeners = stats.current_listeners.saturating_sub(1);
380                if let Some(count) = stats.quality_distribution_now.get_mut(&session.quality) {
381                    *count = count.saturating_sub(1);
382                }
383            }
384
385            debug!("Session terminée: {} ({}ms / {}ms = {:.1}%)", 
386                   session_id, session.duration_played_ms, session.total_duration_ms, session.completion_percentage);
387        }
388    }

390    async fn update_track_analytics(&self, track_id: &str, session: &PlaySession) {
391        let mut analytics = self.track_analytics.write().await;
392        
393        let track_stats = analytics.entry(track_id.to_string()).or_insert_with(|| TrackAnalytics {
394            track_id: track_id.to_string(),
395            total_plays: 0,
396            unique_listeners: 0,
397            total_duration_played_ms: 0,
398            average_completion_rate: 0.0,
399            peak_concurrent_listeners: 0,
400            plays_by_hour: HashMap::new(),
401            plays_by_day: HashMap::new(),
402            skip_rate: 0.0,
403            quality_distribution: HashMap::new(),
404            geographic_distribution: HashMap::new(),
405            platform_distribution: HashMap::new(),
406            last_updated: SystemTime::now(),
407        });
408
409        track_stats.total_plays += 1;
410        track_stats.total_duration_played_ms += session.duration_played_ms;
411        track_stats.average_completion_rate = 
412            (track_stats.average_completion_rate * (track_stats.total_plays - 1) as f32 + session.completion_percentage) 
413            / track_stats.total_plays as f32;
414
415        // Stats par heure
416        let hour = chrono::DateTime::from_timestamp(
417            session.started_at.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64, 0
418        ).unwrap().hour();
419        *track_stats.plays_by_hour.entry(hour).or_insert(0) += 1;
420
421        // Stats par jour
422        let date = chrono::DateTime::from_timestamp(
423            session.started_at.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64, 0
424        ).unwrap().format("%Y-%m-%d").to_string();
425        *track_stats.plays_by_day.entry(date).or_insert(0) += 1;
426
427        // Distribution qualité
428        *track_stats.quality_distribution.entry(session.quality.clone()).or_insert(0) += 1;
429
430        // Distribution plateforme
431        *track_stats.platform_distribution.entry(session.platform.clone()).or_insert(0) += 1;
432
433        track_stats.last_updated = SystemTime::now();
434    }

487    /// Génère un rapport analytics pour une période
488    pub async fn generate_period_report(
489        &self,
490        start_date: SystemTime,
491        end_date: SystemTime,
492    ) -> Result<serde_json::Value, sqlx::Error> {
493        let start_ts = start_date.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64;
494        let end_ts = end_date.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64;
495
496        // Utiliser des requêtes SQL simples sans macros pour éviter les erreurs de driver
497        let total_sessions = sqlx::query("SELECT COUNT(*) as count FROM play_sessions WHERE started_at BETWEEN ? AND ?")
498            .bind(start_ts)
499            .bind(end_ts)
500            .fetch_one(&self.db_pool).await?
501            .get::<i64, _>("count");
502
503        let unique_listeners = sqlx::query("SELECT COUNT(DISTINCT user_id) as count FROM play_sessions WHERE started_at BETWEEN ? AND ?")
504            .bind(start_ts)
505            .bind(end_ts)
506            .fetch_one(&self.db_pool).await?
507            .get::<i64, _>("count");
508
509        let average_completion = sqlx::query("SELECT AVG(completion_percentage) as avg FROM play_sessions WHERE started_at BETWEEN ? AND ?")
510            .bind(start_ts)
511            .bind(end_ts)
512            .fetch_one(&self.db_pool).await?
513            .get::<f64, _>("avg");
514
515        Ok(serde_json::json!({
516            "period": {
517                "start": start_ts,
518                "end": end_ts
519            },
520            "summary": {
521                "total_sessions": total_sessions,
522                "unique_listeners": unique_listeners,
523                "average_completion_rate": average_completion
524            },
525            "top_tracks": []
526        }))
527    }

542    /// Met à jour les statistiques temps réel (à appeler périodiquement)
543    pub async fn refresh_realtime_stats(&self) {
544        let one_hour_ago = SystemTime::now() - std::time::Duration::from_secs(3600);
545        let one_hour_ago_ts = one_hour_ago.duration_since(SystemTime::UNIX_EPOCH).unwrap().as_secs() as i64;
546        
547        // Utiliser des requêtes simples pour éviter les erreurs de compilation
548        let streams_last_hour = sqlx::query("SELECT COUNT(*) as count FROM play_sessions WHERE started_at > ?")
549            .bind(one_hour_ago_ts)
550            .fetch_one(&self.db_pool).await
551            .map(|row| row.get::<i64, _>("count"))
552            .unwrap_or(0);
553
554        let mut stats = self.realtime_stats.write().await;
555        stats.streams_started_last_hour = streams_last_hour as u64;
556        stats.current_listeners = self.active_sessions.read().await.len() as u32;
557    }

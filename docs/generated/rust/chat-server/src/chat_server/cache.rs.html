7/// Cache entry avec expiration
8#[derive(Debug, Clone)]
9pub struct CacheEntry<T> {
10    pub value: T,
11    pub expires_at: Instant,
12    pub hit_count: u64,
13    pub last_accessed: Instant,
14}
15
16impl<T> CacheEntry<T> {
17    pub fn new(value: T, ttl: Duration) -> Self {
18        let now = Instant::now();
19        Self {
20            value,
21            expires_at: now + ttl,
22            hit_count: 0,
23            last_accessed: now,
24        }
25    }
26
27    pub fn is_expired(&self) -> bool {
28        Instant::now() > self.expires_at
29    }
30
31    pub fn touch(&mut self) {
32        self.hit_count += 1;
33        self.last_accessed = Instant::now();
34    }
35}

37/// Cache intelligent avec LRU et expiration
38pub struct SmartCache<K, V> 
39where 
40    K: Clone + std::hash::Hash + Eq,
41    V: Clone,
42{
43    entries: Arc<RwLock<HashMap<K, CacheEntry<V>>>>,
44    max_size: usize,
45    default_ttl: Duration,
46}
47
48impl<K, V> SmartCache<K, V>
49where 
50    K: Clone + std::hash::Hash + Eq,
51    V: Clone,
52{
53    pub fn new(max_size: usize, default_ttl: Duration) -> Self {
54        Self {
55            entries: Arc::new(RwLock::new(HashMap::new())),
56            max_size,
57            default_ttl,
58        }
59    }
60
61    /// Ins√®re une valeur dans le cache
62    pub async fn insert(&self, key: K, value: V) {
63        self.insert_with_ttl(key, value, self.default_ttl).await;
64    }
65
66    /// Ins√®re une valeur avec un TTL personnalis√©
67    pub async fn insert_with_ttl(&self, key: K, value: V, ttl: Duration) {
68        let mut entries = self.entries.write().await;
69        
70        // Nettoyage des entr√©es expir√©es
71        self.cleanup_expired(&mut entries).await;
72        
73        // √âviction LRU si le cache est plein
74        if entries.len() >= self.max_size {
75            self.evict_lru(&mut entries).await;
76        }
77
78        entries.insert(key, CacheEntry::new(value, ttl));
79    }
80
81    /// R√©cup√®re une valeur du cache
82    pub async fn get(&self, key: &K) -> Option<V> {
83        let mut entries = self.entries.write().await;
84        
85        if let Some(entry) = entries.get_mut(key) {
86            if entry.is_expired() {
87                entries.remove(key);
88                return None;
89            }
90            
91            entry.touch();
92            Some(entry.value.clone())
93        } else {
94            None
95        }
96    }
97
98    /// Supprime une entr√©e du cache
99    pub async fn remove(&self, key: &K) -> Option<V> {
100        let mut entries = self.entries.write().await;
101        entries.remove(key).map(|entry| entry.value)
102    }
103
104    /// Nettoie les entr√©es expir√©es
105    async fn cleanup_expired(&self, entries: &mut HashMap<K, CacheEntry<V>>) {
106        let expired_keys: Vec<K> = entries.iter()
107            .filter(|(_, entry)| entry.is_expired())
108            .map(|(key, _)| key.clone())
109            .collect();
110
111        for key in expired_keys {
112            entries.remove(&key);
113        }
114    }
115
116    /// √âviction LRU (Least Recently Used)
117    async fn evict_lru(&self, entries: &mut HashMap<K, CacheEntry<V>>) {
118        if let Some((lru_key, _)) = entries.iter()
119            .min_by_key(|(_, entry)| entry.last_accessed)
120            .map(|(key, entry)| (key.clone(), entry.clone())) {
121            entries.remove(&lru_key);
122        }
123    }
124
125    /// Statistiques du cache
126    pub async fn stats(&self) -> CacheStats {
127        let entries = self.entries.read().await;
128        let total_hits: u64 = entries.values().map(|entry| entry.hit_count).sum();
129        
130        CacheStats {
131            total_entries: entries.len(),
132            max_size: self.max_size,
133            total_hits,
134            hit_rate: if entries.is_empty() { 0.0 } else { total_hits as f64 / entries.len() as f64 },
135        }
136    }
137
138    /// Vide le cache
139    pub async fn clear(&self) {
140        let mut entries = self.entries.write().await;
141        entries.clear();
142    }
143}

3use serde::{Deserialize, Serialize};
4
5#[derive(Debug, Deserialize)]
6#[serde(tag = "type")]
7pub enum WsInbound {
8    #[serde(rename = "join_room")]
9    Join {
10        room: String,
11    },
12
13    #[serde(rename = "room_message")]
14    Message {
15        room: String,
16        content: String,
17    },
18
19    #[serde(rename = "direct_message")]
20    DirectMessage {
21        to_user_id: i32,
22        content: String,
23    },
24
25    #[serde(rename = "room_history")]
26    RoomHistory {
27        room: String,
28        limit: i64,
29    },
30
31    #[serde(rename = "dm_history")]
32    DmHistory {
33        with: i32,
34        limit: i64,
35    }
36}
37
38impl WsInbound {
39    #[allow(dead_code)]
40    pub fn log_received(&self) {
41        match self {
42            WsInbound::Join { room } => {
43                tracing::debug!(message_type = "join_room", room = %room, "游닌 Message join_room re칞u");
44            }
45            WsInbound::Message { room, content } => {
46                tracing::debug!(message_type = "room_message", room = %room, content_length = %content.len(), "游닌 Message room_message re칞u");
47            }
48            WsInbound::DirectMessage { to_user_id, content } => {
49                tracing::debug!(message_type = "direct_message", to_user_id = %to_user_id, content_length = %content.len(), "游닌 Message direct_message re칞u");
50            }
51            WsInbound::RoomHistory { room, limit } => {
52                tracing::debug!(message_type = "room_history", room = %room, limit = %limit, "游닌 Message room_history re칞u");
53            }
54            WsInbound::DmHistory { with, limit } => {
55                tracing::debug!(message_type = "dm_history", with_user = %with, limit = %limit, "游닌 Message dm_history re칞u");
56            }
57        }
58    }
59}

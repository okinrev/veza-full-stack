184    /// Génère une nouvelle paire de clés pour un utilisateur
185    pub async fn generate_user_keypair(&self, user_id: i64) -> Result<UserKeyPair> {
186        let mut public_key = vec![0u8; 32];
187        self.rng.fill(&mut public_key)
188            .map_err(|_| ChatError::internal_error("Failed to generate key"))?;
189        
190        let fingerprint = self.generate_fingerprint(&public_key);
191        
192        let keypair = UserKeyPair {
193            user_id,
194            public_key,
195            fingerprint,
196            status: KeyStatus::Active,
197        };
198        
199        self.user_public_keys.insert(user_id, keypair.clone());
200        Ok(keypair)
201    }

203    /// Établit une session de chiffrement pour un channel
204    pub async fn establish_session(
205        &self,
206        channel_id: String,
207        participants: Vec<i64>,
208        algorithm: EncryptionAlgorithm,
209    ) -> Result<EncryptionSession> {
210        for &user_id in &participants {
211            if !self.user_public_keys.contains_key(&user_id) {
212                return Err(ChatError::validation_error(
213                    &format!("User {} has no public key", user_id)
214                ));
215            }
216        }
217        
218        let session_key = self.generate_session_key()?;
219        let mut encrypted_keys = HashMap::new();
220        
221        for &user_id in &participants {
222            if let Some(user_keypair) = self.user_public_keys.get(&user_id) {
223                let encrypted_key = self.encrypt_session_key(&session_key, &user_keypair.public_key)?;
224                encrypted_keys.insert(user_id, encrypted_key);
225            }
226        }
227        
228        let session = EncryptionSession {
229            session_id: format!("session_{}", uuid::Uuid::new_v4()),
230            channel_id: channel_id.clone(),
231            participants,
232            encrypted_keys,
233            algorithm,
234            status: SessionStatus::Active,
235        };
236        
237        self.encryption_sessions.insert(channel_id, session.clone());
238        Ok(session)
239    }

241    /// Chiffre un message pour un channel
242    pub async fn encrypt_message(
243        &self,
244        channel_id: &str,
245        sender_id: i64,
246        content: &str,
247        message_id: String,
248    ) -> Result<EncryptedMessage> {
249        let session = self.encryption_sessions.get(channel_id)
250            .ok_or_else(|| ChatError::not_found("session", "session_not_found"))?;
251        
252        if !session.participants.contains(&sender_id) {
253            return Err(ChatError::permission_denied("User not authorized"));
254        }
255        
256        let nonce = self.generate_nonce()?;
257        let session_key = &session.encrypted_keys[&sender_id];
258        
259        let (encrypted_content, auth_tag) = self.encrypt_content(
260            content.as_bytes(),
261            session_key,
262            &nonce,
263            &session.algorithm,
264        )?;
265        
266        Ok(EncryptedMessage {
267            message_id,
268            session_id: session.session_id.clone(),
269            encrypted_content,
270            nonce,
271            auth_tag,
272            algorithm: session.algorithm.clone(),
273        })
274    }

276    /// Déchiffre un message
277    pub async fn decrypt_message(
278        &self,
279        encrypted_message: &EncryptedMessage,
280        recipient_id: i64,
281    ) -> Result<String> {
282        let session = self.encryption_sessions.iter()
283            .find(|entry| entry.value().session_id == encrypted_message.session_id)
284            .ok_or_else(|| ChatError::not_found("session", "session_not_found"))?;
285        
286        if !session.participants.contains(&recipient_id) {
287            return Err(ChatError::permission_denied("Not authorized"));
288        }
289        
290        let encrypted_session_key = session.encrypted_keys.get(&recipient_id)
291            .ok_or_else(|| ChatError::not_found("session_key", "key_not_found"))?;
292        
293        let decrypted_content = self.decrypt_content(
294            &encrypted_message.encrypted_content,
295            &encrypted_message.auth_tag,
296            encrypted_session_key,
297            &encrypted_message.nonce,
298            &encrypted_message.algorithm,
299        )?;
300        
301        String::from_utf8(decrypted_content)
302            .map_err(|_| ChatError::internal_error("Invalid decrypted content"))
303    }

305    /// Révoque une session de chiffrement
306    pub async fn revoke_session(&self, channel_id: &str, revoked_by: i64) -> Result<()> {
307        if let Some(mut session) = self.encryption_sessions.get_mut(channel_id) {
308            // Vérifier que l'utilisateur peut révoquer la session
309            if !session.participants.contains(&revoked_by) {
310                return Err(ChatError::permission_denied("Non autorisé à révoquer cette session"));
311            }
312            
313            session.status = SessionStatus::Revoked;
314            tracing::info!("Session {} révoquée par l'utilisateur {}", session.session_id, revoked_by);
315        }
316        
317        Ok(())
318    }

332        if let Some(prefs) = self.user_preferences.get(&user_id) {

364    /// Génère le fingerprint d'une clé publique
365    fn generate_fingerprint(&self, public_key: &[u8]) -> String {
366        use ring::digest;
367        let digest = digest::digest(&digest::SHA256, public_key);
368        BASE64.encode(digest.as_ref())
369    }

371    /// Chiffre une clé de session avec une clé publique
372    fn encrypt_session_key(&self, session_key: &[u8], public_key: &[u8]) -> Result<Vec<u8>> {
373        let mut encrypted = session_key.to_vec();
374        for (i, &byte) in public_key.iter().enumerate() {
375            if i < encrypted.len() {
376                encrypted[i] ^= byte;
377            }
378        }
379        Ok(encrypted)
380    }

382    /// Chiffre du contenu avec AES-GCM
383    fn encrypt_content(
384        &self,
385        content: &[u8],
386        key: &[u8],
387        nonce: &[u8],
388        algorithm: &EncryptionAlgorithm,
389    ) -> Result<(Vec<u8>, Vec<u8>)> {
390        match algorithm {
391            EncryptionAlgorithm::AES256GCM => self.encrypt_aes_gcm(content, key, nonce),
392            EncryptionAlgorithm::ChaCha20Poly1305 => self.encrypt_chacha20_poly1305(content, key, nonce),
393        }
394    }

399        encrypted_content: &[u8],
400        auth_tag: &[u8],
401        key: &[u8],
402        nonce: &[u8],
403        algorithm: &EncryptionAlgorithm,

405        match algorithm {

407            EncryptionAlgorithm::ChaCha20Poly1305 => self.decrypt_chacha20_poly1305(encrypted_content, auth_tag, key, nonce),
408        }

411    /// Chiffrement AES-256-GCM
412    fn encrypt_aes_gcm(&self, content: &[u8], key: &[u8], nonce: &[u8]) -> Result<(Vec<u8>, Vec<u8>)> {
413        let unbound_key = aead::UnboundKey::new(&aead::AES_256_GCM, key)
414            .map_err(|_| ChatError::internal_error("Invalid AES key"))?;
415        
416        let key = aead::LessSafeKey::new(unbound_key);
417        let nonce = aead::Nonce::try_assume_unique_for_key(nonce)
418            .map_err(|_| ChatError::internal_error("Invalid nonce"))?;
419        
420        let mut in_out = content.to_vec();
421        let tag = key.seal_in_place_separate_tag(nonce, aead::Aad::empty(), &mut in_out)
422            .map_err(|_| ChatError::internal_error("Encryption failed"))?;
423        
424        Ok((in_out, tag.as_ref().to_vec()))
425    }

427    /// Déchiffrement AES-256-GCM
428    fn decrypt_aes_gcm(
429        &self,
430        encrypted_content: &[u8],
431        auth_tag: &[u8],
432        key: &[u8],
433        nonce: &[u8],
434    ) -> Result<Vec<u8>> {
435        let unbound_key = aead::UnboundKey::new(&aead::AES_256_GCM, key)
436            .map_err(|_| ChatError::internal_error("Invalid AES key"))?;
437        
438        let key = aead::LessSafeKey::new(unbound_key);
439        let nonce = aead::Nonce::try_assume_unique_for_key(nonce)
440            .map_err(|_| ChatError::internal_error("Invalid nonce"))?;
441        
442        let mut in_out = encrypted_content.to_vec();
443        in_out.extend_from_slice(auth_tag);
444        
445        let plaintext = key.open_in_place(nonce, aead::Aad::empty(), &mut in_out)
446            .map_err(|_| ChatError::internal_error("Decryption failed"))?;
447        
448        Ok(plaintext.to_vec())
449    }

264    /// Vérifie si une requête est autorisée (point d'entrée principal)
265    pub async fn check_rate_limit(
266        &self,
267        ip: IpAddr,
268        user_id: Option<i64>,
269        channel_id: Option<String>,
270        _limit_type: LimitType,
271        request_info: RequestInfo,
272    ) -> Result<RateLimitResult> {
273        let start_time = Instant::now();
274        
275        // Incrémenter les métriques
276        self.metrics.requests_processed.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
277        
278        // 1. Vérifier la liste noire d'abord
279        if let Some(blacklist_entry) = self.blacklist.get(&ip) {
280            if blacklist_entry.expires_at.map_or(true, |exp| exp > Instant::now()) {
281                self.metrics.requests_blocked.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
282                return Ok(RateLimitResult {
283                    allowed: false,
284                    reason: Some(format!("IP blacklisted: {}", blacklist_entry.reason)),
285                    retry_after: blacklist_entry.expires_at.map(|exp| exp.duration_since(Instant::now())),
286                    remaining_tokens: 0,
287                    burst_remaining: 0,
288                    reputation_impact: -0.1,
289                });
290            } else {
291                // Entrée expirée, la supprimer
292                self.blacklist.remove(&ip);
293            }
294        }
295        
296        // 2. Vérifier le rate limiting par IP
297        let ip_result = self.check_ip_rate_limit(ip, &_limit_type, &request_info).await?;
298        if !ip_result.allowed {
299            return Ok(ip_result);
300        }
301        
302        // 3. Vérifier le rate limiting par utilisateur si applicable
303        if let Some(uid) = user_id {
304            let user_result = self.check_user_rate_limit(uid, &_limit_type).await?;
305            if !user_result.allowed {
306                return Ok(user_result);
307            }
308        }
309        
310        // 4. Vérifier le rate limiting par canal si applicable
311        if let Some(cid) = channel_id {
312            let channel_result = self.check_channel_rate_limit(&cid, &_limit_type).await?;
313            if !channel_result.allowed {
314                return Ok(channel_result);
315            }
316        }
317        
318        // 5. Analyser les patterns d'attaque
319        self.analyze_request_pattern(ip, &request_info).await?;
320        
321        // 6. Mettre à jour les métriques de performance
322        let elapsed = start_time.elapsed();
323        *self.metrics.avg_response_time.lock() = elapsed;
324        
325        Ok(RateLimitResult {
326            allowed: true,
327            reason: None,
328            retry_after: None,
329            remaining_tokens: ip_result.remaining_tokens,
330            burst_remaining: ip_result.burst_remaining,
331            reputation_impact: 0.0,
332        })
333    }

335    /// Vérifie le rate limiting par IP
336    async fn check_ip_rate_limit(
337        &self,
338        ip: IpAddr,
339        _limit_type: &LimitType,
340        request_info: &RequestInfo,
341    ) -> Result<RateLimitResult> {
342        let config = self.config.read().await;
343        
344        // Récupérer ou créer le limiteur IP
345        let mut ip_limiter = self.ip_limiters.entry(ip).or_insert_with(|| {
346            IpRateLimiter::new(ip, &config)
347        });
348        
349        // Vérifier le statut de l'IP
350        match ip_limiter.status {
351            IpStatus::Banned => {
352                return Ok(RateLimitResult {
353                    allowed: false,
354                    reason: Some("IP permanently banned".to_string()),
355                    retry_after: None,
356                    remaining_tokens: 0,
357                    burst_remaining: 0,
358                    reputation_impact: -0.2,
359                });
360            }
361            IpStatus::Blacklisted => {
362                return Ok(RateLimitResult {
363                    allowed: false,
364                    reason: Some("IP temporarily blacklisted".to_string()),
365                    retry_after: Some(Duration::from_secs(300)), // 5 minutes
366                    remaining_tokens: 0,
367                    burst_remaining: 0,
368                    reputation_impact: -0.1,
369                });
370            }
371            _ => {}
372        }
373        
374        // Appliquer le rate limiting avec token bucket
375        let remaining_tokens = {
376            let bucket = ip_limiter.buckets.get_mut(_limit_type).unwrap();
377            bucket.refill();
378            
379            if bucket.tokens > 0 {
380                bucket.tokens -= 1;
381                bucket.tokens
382            } else {
383                0
384            }
385        };
386        
387        if remaining_tokens > 0 {
388            ip_limiter.last_activity = Instant::now();
389            
390            // Enregistrer l'événement pour l'analyse de patterns
391            ip_limiter.request_patterns.push_back(RequestEvent {
392                timestamp: Instant::now(),
393                request_type: format!("{:?}", _limit_type),
394                path: request_info.path.clone(),
395                user_agent: request_info.user_agent.clone(),
396                response_time: Duration::from_millis(0),
397                status_code: 200,
398            });
399            
400            // Garder seulement les 100 derniers événements
401            if ip_limiter.request_patterns.len() > 100 {
402                ip_limiter.request_patterns.pop_front();
403            }
404            
405            Ok(RateLimitResult {
406                allowed: true,
407                reason: None,
408                retry_after: None,
409                remaining_tokens,
410                burst_remaining: remaining_tokens,
411                reputation_impact: 0.0,
412            })
413        } else {
414            // Rate limit dépassé
415            ip_limiter.violation_count += 1;
416            
417            // Escalade automatique si trop de violations
418            if ip_limiter.violation_count >= 5 {
419                ip_limiter.status = IpStatus::Suspicious;
420            }
421            if ip_limiter.violation_count >= 10 {
422                self.auto_blacklist_ip(ip, "Too many violations".to_string()).await?;
423            }
424            
425            Ok(RateLimitResult {
426                allowed: false,
427                reason: Some(format!("Rate limit exceeded for {:?}", _limit_type)),
428                retry_after: Some(Duration::from_secs(60)),
429                remaining_tokens: 0,
430                burst_remaining: 0,
431                reputation_impact: 0.0,
432            })
433        }
434    }

436    /// Vérifie le rate limiting par utilisateur
437    async fn check_user_rate_limit(&self, user_id: i64, _limit_type: &LimitType) -> Result<RateLimitResult> {
438        let config = self.config.read().await;
439        
440        let mut user_limiter = self.user_limiters.entry(user_id).or_insert_with(|| {
441            UserRateLimiter::new(user_id, &config)
442        });
443        
444        // Vérifier la réputation d'abord
445        let capacity_multiplier = match user_limiter.reputation.level {
446            ReputationLevel::NewUser => 0.5,
447            ReputationLevel::Normal => 1.0,
448            ReputationLevel::Trusted => 1.5,
449            ReputationLevel::VIP => 2.0,
450            ReputationLevel::System => 5.0,
451        };
452
453        // Puis accéder au bucket avec la capacité ajustée
454        let remaining_tokens = {
455            let bucket = user_limiter.buckets.get_mut(_limit_type).unwrap();
456            bucket.capacity = (bucket.capacity as f32 * capacity_multiplier) as u32;
457            bucket.refill();
458            
459            if bucket.tokens > 0 {
460                bucket.tokens -= 1;
461                bucket.tokens
462            } else {
463                0
464            }
465        };
466        
467        if remaining_tokens > 0 {
468            user_limiter.last_activity = Instant::now();
469            
470            Ok(RateLimitResult {
471                allowed: true,
472                reason: None,
473                retry_after: None,
474                remaining_tokens,
475                burst_remaining: remaining_tokens,
476                reputation_impact: 0.0,
477            })
478        } else {
479            user_limiter.violation_count += 1;
480            user_limiter.reputation.violations_today += 1;
481            user_limiter.reputation.score = (user_limiter.reputation.score - 0.05).max(0.0);
482            
483            Ok(RateLimitResult {
484                allowed: false,
485                reason: Some(format!("User rate limit exceeded for {:?}", _limit_type)),
486                retry_after: Some(Duration::from_secs(30)),
487                remaining_tokens: 0,
488                burst_remaining: 0,
489                reputation_impact: 0.0,
490            })
491        }
492    }

494    /// Vérifie le rate limiting par canal
495    async fn check_channel_rate_limit(&self, channel_id: &str, _limit_type: &LimitType) -> Result<RateLimitResult> {
496        tracing::debug!(?_limit_type, "Vérification du rate limit canal");
497        let config = self.config.read().await;
498        
499        let mut channel_limiter = self.channel_limiters.entry(channel_id.to_string()).or_insert_with(|| {
500            ChannelRateLimiter::new(channel_id.to_string(), &config)
501        });
502        
503        // Appliquer la modération selon le niveau du canal et le type de limite
504        let rate_multiplier = match (channel_limiter.moderation_level.clone(), _limit_type) {
505            (ModerationLevel::Low, LimitType::MessagesPerMinute) => 1.5,
506            (ModerationLevel::Normal, LimitType::MessagesPerMinute) => 1.0,
507            (ModerationLevel::High, LimitType::MessagesPerMinute) => 0.5,
508            (ModerationLevel::Lockdown, LimitType::MessagesPerMinute) => 0.1,
509            (_, _) => 1.0, // Valeur par défaut pour autres types
510        };
511        
512        channel_limiter.message_bucket.refill();
513        let tokens_needed = (1.0 / rate_multiplier) as u32;
514        
515        if channel_limiter.message_bucket.tokens >= tokens_needed {
516            channel_limiter.message_bucket.tokens -= tokens_needed;
517            channel_limiter.last_activity = Instant::now();
518            
519            Ok(RateLimitResult {
520                allowed: true,
521                reason: None,
522                retry_after: None,
523                remaining_tokens: channel_limiter.message_bucket.tokens,
524                burst_remaining: channel_limiter.message_bucket.burst_allowance,
525                reputation_impact: 0.0,
526            })
527        } else {
528            Ok(RateLimitResult {
529                allowed: false,
530                reason: Some(format!("Channel rate limit exceeded (moderation: {:?})", channel_limiter.moderation_level)),
531                retry_after: Some(Duration::from_secs(10)),
532                remaining_tokens: 0,
533                burst_remaining: 0,
534                reputation_impact: 0.0,
535            })
536        }
537    }

539    /// Analyse les patterns de requêtes pour détecter les attaques
540    async fn analyze_request_pattern(&self, ip: IpAddr, request_info: &RequestInfo) -> Result<()> {
541        // Récupérer l'historique des requêtes pour cette IP
542        if let Some(ip_limiter) = self.ip_limiters.get(&ip) {
543            let recent_requests: Vec<_> = ip_limiter.request_patterns.iter()
544                .filter(|event| event.timestamp.elapsed() < Duration::from_secs(60))
545                .collect();
546            
547            // Détecter différents types d'attaques
548            
549            // 1. DDoS - Trop de requêtes dans un court laps de temps
550            if recent_requests.len() > 100 {
551                self.detect_ddos_attack(ip, &recent_requests).await?;
552            }
553            
554            // 2. Brute Force - Tentatives répétées sur l'authentification
555            if request_info.path.contains("/auth") || request_info.path.contains("/login") {
556                let auth_attempts = recent_requests.iter()
557                    .filter(|event| event.path.contains("/auth"))
558                    .count();
559                
560                if auth_attempts > 10 {
561                    self.detect_brute_force_attack(ip).await?;
562                }
563            }
564            
565            // 3. Bot Activity - Patterns suspects (même User-Agent, timing régulier)
566            if let Some(user_agent) = &request_info.user_agent {
567                let same_ua_count = recent_requests.iter()
568                    .filter(|event| event.user_agent.as_ref() == Some(user_agent))
569                    .count();
570                
571                if same_ua_count > 50 && self.is_suspicious_user_agent(user_agent) {
572                    self.detect_bot_activity(ip, user_agent.clone()).await?;
573                }
574            }
575        }
576        
577        Ok(())
578    }

580    /// Détecte une attaque DDoS
581    async fn detect_ddos_attack(&self, ip: IpAddr, recent_requests: &[&RequestEvent]) -> Result<()> {
582        let pattern = AttackPattern {
583            pattern_id: format!("ddos_{}_{}",ip, Instant::now().elapsed().as_secs()),
584            pattern_type: AttackType::DDoS,
585            source_ips: vec![ip],
586            detection_time: Instant::now(),
587            severity: 0.9,
588            requests_count: recent_requests.len() as u32,
589            geographic_spread: 0.0, // Single IP
590            user_agents: recent_requests.iter()
591                .filter_map(|event| event.user_agent.clone())
592                .collect::<std::collections::HashSet<_>>()
593                .into_iter()
594                .collect(),
595        };
596        
597        self.attack_patterns.insert(pattern.pattern_id.clone(), pattern);
598        self.auto_blacklist_ip(ip, "DDoS attack detected".to_string()).await?;
599        
600        self.metrics.attacks_detected.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
601        tracing::warn!("DDoS attack detected from IP: {}", ip);
602        
603        Ok(())
604    }

626    /// Détecte une activité de bot
627    async fn detect_bot_activity(&self, ip: IpAddr, user_agent: String) -> Result<()> {
628        let pattern = AttackPattern {
629            pattern_id: format!("bot_{}_{}", ip, Instant::now().elapsed().as_secs()),
630            pattern_type: AttackType::BotActivity,
631            source_ips: vec![ip],
632            detection_time: Instant::now(),
633            severity: 0.6,
634            requests_count: 50,
635            geographic_spread: 0.0,
636            user_agents: vec![user_agent],
637        };
638        
639        self.attack_patterns.insert(pattern.pattern_id.clone(), pattern);
640        
641        // Marquer l'IP comme suspecte plutôt que de la blacklister immédiatement
642        if let Some(mut ip_limiter) = self.ip_limiters.get_mut(&ip) {
643            ip_limiter.status = IpStatus::Suspicious;
644            ip_limiter.trust_score = (ip_limiter.trust_score - 0.3).max(0.0);
645        }
646        
647        tracing::info!("Bot activity detected from IP: {}", ip);
648        Ok(())
649    }

651    /// Vérifie si un User-Agent est suspect
652    fn is_suspicious_user_agent(&self, user_agent: &str) -> bool {
653        let suspicious_patterns = [
654            "bot", "crawler", "spider", "scraper", "curl", "wget", "python", "java", 
655            "headless", "selenium", "phantom", "automated"
656        ];
657        
658        let ua_lower = user_agent.to_lowercase();
659        suspicious_patterns.iter().any(|pattern| ua_lower.contains(pattern))
660    }

662    /// Ajoute automatiquement une IP à la liste noire
663    async fn auto_blacklist_ip(&self, ip: IpAddr, reason: String) -> Result<()> {
664        let reason_clone = reason.clone();
665        let config = self.config.read().await;
666        
667        let blacklist_entry = BlacklistEntry {
668            ip,
669            reason,
670            blocked_at: Instant::now(),
671            expires_at: Some(Instant::now() + config.auto_blacklist_duration),
672            violation_count: 1,
673            auto_generated: true,
674        };
675        
676        self.blacklist.insert(ip, blacklist_entry);
677        self.metrics.attacks_detected.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
678        
679        tracing::warn!("IP {} automatically blacklisted: {}", ip, reason_clone);
680        Ok(())
681    }
682    
683    /// Nettoie les entrées expirées
684    pub async fn cleanup_expired_entries(&self) {
685        let now = Instant::now();
686        
687        // Nettoyer les blacklists expirées
688        self.blacklist.retain(|_, entry| {
689            entry.expires_at.map_or(true, |exp| exp > now)
690        });
691        
692        // Nettoyer les limiteurs inactifs (plus de 1 heure)
693        self.ip_limiters.retain(|_, limiter| {
694            now.duration_since(limiter.last_activity) < Duration::from_secs(3600)
695        });
696        
697        self.user_limiters.retain(|_, limiter| {
698            now.duration_since(limiter.last_activity) < Duration::from_secs(3600)
699        });
700        
701        self.channel_limiters.retain(|_, limiter| {
702            now.duration_since(limiter.last_activity) < Duration::from_secs(3600)
703        });
704    }
705}

716impl TokenBucket {
717    pub fn new(capacity: u32, refill_rate: f32, burst_allowance: u32) -> Self {
718        Self {
719            capacity,
720            tokens: capacity,
721            refill_rate,
722            last_refill: Instant::now(),
723            burst_allowance,
724        }
725    }
726    
727    pub fn refill(&mut self) {
728        let now = Instant::now();
729        let elapsed = now.duration_since(self.last_refill).as_secs_f32();
730        let tokens_to_add = (elapsed * self.refill_rate) as u32;
731        
732        if tokens_to_add > 0 {
733            self.tokens = (self.tokens + tokens_to_add).min(self.capacity);
734            self.last_refill = now;
735        }
736    }
737}

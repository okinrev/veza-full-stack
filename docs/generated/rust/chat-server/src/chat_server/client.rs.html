3use tokio::sync::mpsc::UnboundedSender;
4use tokio_tungstenite::tungstenite::Message;
5use std::time::{Duration, Instant};
6
7#[derive(Debug, Clone)]
8pub struct Client {
9    pub user_id: i32,
10    pub username: String,
11    pub sender: UnboundedSender<Message>,
12    pub last_heartbeat: std::sync::Arc<std::sync::RwLock<Instant>>,
13    pub connected_at: Instant,
14}
15
16impl Client {
17    pub fn new(user_id: i32, username: String, sender: UnboundedSender<Message>) -> Self {
18        Self {
19            user_id,
20            username,
21            sender,
22            last_heartbeat: std::sync::Arc::new(std::sync::RwLock::new(Instant::now())),
23            connected_at: Instant::now(),
24        }
25    }
26
27    /// Envoie un message texte au client
28    pub fn send_text(&self, text: &str) -> bool {
29        tracing::debug!(user_id = %self.user_id, username = %self.username, text_length = %text.len(), "🔧 Tentative d'envoi de message texte");
30        
31        match self.sender.send(Message::Text(text.to_string())) {
32            Ok(_) => {
33                tracing::debug!(user_id = %self.user_id, username = %self.username, "✅ Message texte envoyé au canal");
34                true
35            }
36            Err(e) => {
37                tracing::error!(user_id = %self.user_id, username = %self.username, error = %e, "❌ Erreur envoi message texte au canal");
38                false
39            }
40        }
41    }
42
43    /// Envoie un ping pour vérifier la connexion
44    pub fn send_ping(&self) -> bool {
45        tracing::debug!(user_id = %self.user_id, username = %self.username, "🏓 Envoi ping");
46        
47        match self.sender.send(Message::Ping(vec![])) {
48            Ok(_) => {
49                tracing::debug!(user_id = %self.user_id, username = %self.username, "✅ Ping envoyé");
50                true
51            }
52            Err(e) => {
53                tracing::error!(user_id = %self.user_id, username = %self.username, error = %e, "❌ Erreur envoi ping");
54                false
55            }
56        }
57    }
58
59    /// Met à jour le timestamp du dernier heartbeat
60    pub fn update_heartbeat(&self) {
61        if let Ok(mut last_heartbeat) = self.last_heartbeat.write() {
62            *last_heartbeat = Instant::now();
63        }
64    }
65
66    /// Vérifie si la connexion est encore active (basé sur le heartbeat)
67    pub fn is_alive(&self, timeout: Duration) -> bool {
68        if let Ok(last_heartbeat) = self.last_heartbeat.read() {
69            last_heartbeat.elapsed() < timeout
70        } else {
71            false
72        }
73    }
74
75    /// Retourne la durée de connexion
76    pub fn connection_duration(&self) -> Duration {
77        self.connected_at.elapsed()
78    }
79}

1//! Store de messages simple et fonctionnel
2
3use chrono::{DateTime, Utc};
4use serde::{Serialize, Deserialize};
5use tokio::sync::RwLock;
6use crate::error::{ChatError, Result};
7
8
9/// Message simple pour test
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct SimpleMessage {
12    pub id: i32,
13    pub content: String,
14    pub author: String,
15    pub timestamp: DateTime<Utc>,
16    pub room: Option<String>,
17    pub is_direct: bool,
18}
19
20/// Store en mémoire pour les tests
21pub struct SimpleMessageStore {
22    messages: RwLock<Vec<SimpleMessage>>,
23    next_id: RwLock<i32>,
24}
25
26impl SimpleMessageStore {
27    pub fn new() -> Self {
28        Self {
29            messages: RwLock::new(Vec::new()),
30            next_id: RwLock::new(1),
31        }
32    }
33
34    /// Envoi d'un message simple
35    pub async fn send_simple_message(
36        &self,
37        content: &str,
38        author: &str,
39        room: Option<&str>,
40        is_direct: bool,
41    ) -> Result<i32> {
42        let mut next_id = self.next_id.write().await;
43        let id = *next_id;
44        *next_id += 1;
45
46        let message = SimpleMessage {
47            id,
48            content: content.to_string(),
49            author: author.to_string(),
50            timestamp: Utc::now(),
51            room: room.map(|s| s.to_string()),
52            is_direct,
53        };
54
55        let mut messages = self.messages.write().await;
56        messages.push(message);
57
58        Ok(id)
59    }
60
61    /// Récupération des messages d'un salon
62    pub async fn get_room_messages(&self, room_name: &str, limit: i32) -> Result<Vec<SimpleMessage>> {
63        let messages = self.messages.read().await;
64        let filtered: Vec<SimpleMessage> = messages
65            .iter()
66            .filter(|msg| {
67                if let Some(ref msg_room) = msg.room {
68                    msg_room == room_name && !msg.is_direct
69                } else {
70                    false
71                }
72            })
73            .take(limit as usize)
74            .cloned()
75            .collect();
76
77        Ok(filtered)
78    }
79
80    /// Récupération des messages directs
81    pub async fn get_direct_messages(&self, user1: &str, user2: &str, limit: i32) -> Result<Vec<SimpleMessage>> {
82        let messages = self.messages.read().await;
83        let filtered: Vec<SimpleMessage> = messages
84            .iter()
85            .filter(|msg| {
86                msg.is_direct && 
87                ((msg.author == user1) || (msg.author == user2))
88            })
89            .take(limit as usize)
90            .cloned()
91            .collect();
92
93        Ok(filtered)
94    }
95
96    /// Autres méthodes simplifiées
97    pub async fn pin_message(&self, _message_id: i32) -> Result<()> { Ok(()) }
98    pub async fn message_exists(&self, message_id: i32) -> Result<bool> { 
99        let messages = self.messages.read().await;
100        Ok(messages.iter().any(|msg| msg.id == message_id))
101    }
102    pub async fn delete_message(&self, message_id: i32) -> Result<()> {
103        let mut messages = self.messages.write().await;
104        messages.retain(|msg| msg.id != message_id);
105        Ok(())
106    }
107    pub async fn edit_message(&self, message_id: i32, new_content: &str) -> Result<()> {
108        let mut messages = self.messages.write().await;
109        if let Some(msg) = messages.iter_mut().find(|msg| msg.id == message_id) {
110            msg.content = new_content.to_string();
111            Ok(())
112        } else {
113            Err(ChatError::not_found("message", &message_id.to_string()))
114        }
115    }
116    pub async fn add_reaction(&self, _message_id: i32, _user_id: i32, _emoji: &str) -> Result<()> { Ok(()) }
117    pub async fn remove_reaction(&self, _message_id: i32, _user_id: i32, _emoji: &str) -> Result<()> { Ok(()) }
118    pub async fn mark_as_read(&self, _user_id: i32, _conversation_id: &str) -> Result<()> { Ok(()) }
119    pub async fn count_unread(&self, _user_id: i32) -> Result<i64> { Ok(0) }
120    pub async fn count_unread_dms(&self, _user_id: i32) -> Result<i64> { Ok(0) }
121    pub async fn count_unread_mentions(&self, _user_id: i32) -> Result<i64> { Ok(0) }
122    pub async fn count_reactions(&self, _message_id: i32) -> Result<i64> { Ok(0) }
123}
